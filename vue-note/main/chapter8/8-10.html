# 8.10 router-link组件

  这个小节我们来分析router-link组件，router-link组件代码是比较多的，但是它并不复杂，相比于router-view组件它要简单的多。router-link组件是一个普通的组件，它定义比较长，所以我们拆开来分析它。

  name: 'RouterLink',
  props: {
    to: {
      type: toTypes,
      required: true
    },
    tag: {
      type: String,
      default: 'a'
    },
    custom: Boolean,
    exact: Boolean,
    exactPath: Boolean,
    append: Boolean,
    replace: Boolean,
    activeClass: String,
    exactActiveClass: String,
    ariaCurrentValue: {
      type: String,
      default: 'page'
    },
    event: {
      type: eventTypes,
      default: 'click'
    }
  },

  先看它的props配置，有我们的to配置，是必传的。至于它们有什么作用直接看官方文档就好了，就不一一举例了。接下来其实就render函数了，我们一步步来看：

  const router = this.$router
  const current = this.$route
  const { location, route, href } = router.resolve(
    this.to,
    current,
    this.append
  )

  获取router和current，然后执行router.resolve拿到location、route、href。然后我看router.resolve方法：

  resolve (
    to: RawLocation,
    current?: Route,
    append?: boolean
  ): {
    location: Location,
    route: Route,
    href: string,
    // for backwards compat
    normalizedTo: Location,
    resolved: Route
  } {
    current = current || this.history.current
    const location = normalizeLocation(to, current, append, this)
    const route = this.match(location, current)
    const fullPath = route.redirectedFrom || route.fullPath
    const base = this.history.base
    const href = createHref(base, fullPath, this.mode)
    return {
      location,
      route,
      href,
      // for backwards compat
      normalizedTo: location,
      resolved: route
    }
  }

  这里第三个参数append，如果为true的话则会将to作为一个相对路径处理。先是处理current，current是当前路径，比如说/路径下有个router-link组件配置是/about，那么current就是/路径对应的route，然后标准化location，这个location就是/about对应的内容。然后通过location去match匹配生成/about的route，然后获取fullPath、base和href，这里fullPath就是/about，我们来看一下base是什么

this.base = normalizeBase(base)

new History时this.base被赋值为normalizeBase，其中base是在options中可配选项

function normalizeBase (base: ?string): string {
  if (!base) {
    if (inBrowser) {
      // respect <base> tag
      const baseEl = document.querySelector('base')
      base = (baseEl && baseEl.getAttribute('href')) || '/'
      // strip full URL origin
      base = base.replace(/^https?:\/\/[^\/]+/, '')
    } else {
      base = '/'
    }
  }
  // make sure there's the starting slash
  if (base.charAt(0) !== '/') {
    base = '/' + base
  }
  // remove trailing slash
  return base.replace(/\/$/, '')
}

  normalizeBase首先判断base如果不存在就执行if逻辑，如果是浏览器环境就去找base原始，如果找到base元素就把base设置其href属性值，否则设置为/。如果base是一个href值的话就把域名给截取掉。如果不是浏览器环境就直接设置为/。 接下来是确保base开头是/，最后去掉尾斜杠返回。然后href的获取则是执行createHref方法

function createHref (base: string, fullPath: string, mode) {
  var path = mode === 'hash' ? '#' + fullPath : fullPath
  return base ? cleanPath(base + '/' + path) : path
}

  这个方法定义非常简单，如果是hash模式则使path为# + fullPath，否则直接为fullPath。最后如果base存在就将base拼接/和path并经过cleanPath处理返回，否则直接返回path，cleanPath是将双斜杠转换为单斜杠，之前已经分析过了。所以这里href最后返回的就是#/about
  最后就是把定义的这些变量返回，然后接着往下看render函数

  const globalActiveClass = router.options.linkActiveClass
  const globalExactActiveClass = router.options.linkExactActiveClass
  // Support global empty active class
  const activeClassFallback =
    globalActiveClass == null ? 'router-link-active' : globalActiveClass
  const exactActiveClassFallback =
    globalExactActiveClass == null
      ? 'router-link-exact-active'

  const activeClass =
    this.activeClass == null ? activeClassFallback : this.activeClass

    this.exactActiveClass == null

处理class

  const compareTarget = route.redirectedFrom
    ? createRoute(null, normalizeLocation(route.redirectedFrom), null, router)
    : route

  classes[exactActiveClass] = isSameRoute(current, compareTarget, this.exactPath)
  classes[activeClass] = this.exact || this.exactPath
    ? classes[exactActiveClass]
    : isIncludedRoute(current, compareTarget)
  const  = classes[exactActiveClass] ? this.ariaCurrentValue : null

  对重定向路由处理，如果current和compareTarget满足isSameRoute也就是全匹配情况下就将其 classes[exactActiveClass]设置为true，exactActiveClass就是全匹配时的类名。classes[activeClass]则是判断this.exact或this.exactPath为真就设置为classes[exactActiveClass]，否则设置为isIncludedRoute返回值

export function isIncludedRoute (current: Route, target: Route): boolean {
  return (
    current.path.replace(trailingSlashRE, '/').indexOf(
      target.path.replace(trailingSlashRE, '/')
    ) === 0 &&
    (!target.hash || current.hash === target.hash) &&
    queryIncludes(current.query, target.query)
  )
}

  判断current.path和target.path开头是否一致，并且(target.hash不存在或者两个hash值相同)并且query有相同key

function queryIncludes (current: Dictionary<string>, target: Dictionary<string>): boolean {
  for (const key in target) {
    if (!(key in current)) {
      return false
    }
  }
  return true
}

  满足这些条件就认为是包含关系。其实就是模糊匹配的意思，正常是模糊匹配，如果设置了exact或者exactPath的话就编程精确匹配了。

  const handler = e => {
    if (guardEvent(e)) {
      if (this.replace) {
        router.replace(location, noop)
      } else {
        router.push(location, noop)
      }

  const on = { click: guardEvent }

  处理一些事件，guardEvent是事件保护，如果一些不合法的事件将不触发，代码如下所示

export function guardEvent (e: any) {
  // don't redirect with control keys
  if (e.metaKey || e.altKey || e.ctrlKey || e.shiftKey) return
  // don't redirect when preventDefault called
  if (e.defaultPrevented) return
  // don't redirect on right click
  if (e.button !== undefined && e.button !== 0) return
  // don't redirect if `target="_blank"`
  if (e.currentTarget && e.currentTarget.getAttribute) {
    const target = e.currentTarget.getAttribute('target')
    if (/\b_blank\b/i.test(target)) return
  }
  // this may be a Weex event which doesn't have this method
  if (e.preventDefault) {
    e.preventDefault()
  }
  return true
}

  比如一些键被按下就不触发事件，preventDefault()被调用的话也不触发事件，如果是鼠标右键点击也不触发事件，如果target设置为_blank也不触发事件。

  const data: any = { class: classes }

  定义data，并将classes对象放到class属性中，然后有关插槽的内容我们就不细说了，接下来看对标签的处理

  if (this.tag === 'a') {
    data.on = on
    data.attrs = { href, 'aria-current': ariaCurrentValue }
  } else {
    // find the first <a> child and apply listener and href
    const a = findAnchor(this.$slots.default)
    if (a) {
      // in case the <a> is a static node
      a.isStatic = false
      const aData = (a.data = extend({}, a.data))
      aData.on = aData.on || {}
      // transform existing events in both objects into arrays so we can push later
      for (const event in aData.on) {
        const handler = aData.on[event]
        if (event in on) {
          aData.on[event] = Array.isArray(handler) ? handler : [handler]
        }
      }
      // append new listeners for router-link
      for (const event in on) {
        if (event in aData.on) {
          // on[event] is always a function
          aData.on[event].push(on[event])
        } else {
          aData.on[event] = handler
        }
      }

      const aAttrs = (a.data.attrs = extend({}, a.data.attrs))
      aAttrs.href = href
      aAttrs['aria-current'] = ariaCurrentValue
    } else {
      // doesn't have <a> child, apply listener to self
      data.on = on
    }
  }

  router-link会被默认处理为a标签，所以如果是默认标签的话就很简单了，只需要把on赋值给data.on，然后设置标签href属性以及aria-current属性。
  如果不是默认标签就去找第一个子的a标签，我们来看findAnchor方法

function findAnchor (children) {
  if (children) {
    let child
    for (let i = 0; i < children.length; i++) {
      child = children[i]
      if (child.tag === 'a') {
        return child
      }
      if (child.children && (child = findAnchor(child.children))) {
        return child
      }
    }
  }
}

  就是去找默认插槽中有没有含有a标签，有的话就直接返回，没有就继续递归往下找。最后找到就返回，找不到就返回undefined。
  如果没有找到a标签，那么处理就很简单了，直接把data.on赋值为on，只要添加事件就可以了，如果找到a标签了，处理看起来就复杂的多了，其实就是把事件做了个合并处理，然后将a标签的href和aria-current重新赋值。

  return h(this.tag, data, this.$slots.default)

  最后通过h函数创建vnode并返回。这就是router-link的所有内容了。

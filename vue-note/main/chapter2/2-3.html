# 2.3 Vue挂载时的mountComponent

export function mountComponent (
  vm: Component,
  el: ?Element,
  hydrating?: boolean
): Component {
  vm.$el = el
  if (!vm.$options.render) {
    vm.$options.render = createEmptyVNode
    if (process.env.NODE_ENV !== 'production') {
      /* istanbul ignore if */
      if ((vm.$options.template && vm.$options.template.charAt(0) !== '#') || vm.$options.el || el) {
        warn(
          'You are using the runtime-only build of Vue where the template ' + 'compiler is not available. Either pre-compile the templates into ' + 'render functions, or use the compiler-included build.',
          vm
        )
      } else {
        warn(
          'Failed to mount component: template or render function not defined.',
          vm
        )
      }
    }
  }
  callHook(vm, 'beforeMount')

  let updateComponent
  /* istanbul ignore if */
  if (process.env.NODE_ENV !== 'production' && config.performance && mark) { // mark 是性能埋点
    ...
  } else {
    updateComponent = () => {
      vm._update(vm._render(), hydrating)  // hydrating 服务端渲染相关
    }
  }
  // we set this to vm._watcher inside the watcher's constructor
  // since the watcher's initial patch may call $forceUpdate (e.g. inside child
  // component's mounted hook), which relies on vm._watcher being already defined
  new Watcher(vm, updateComponent, noop, {
    before () {
      if (vm._isMounted && !vm._isDestroyed) {
        callHook(vm, 'beforeUpdate')
      }
    }
  }, true /* isRenderWatcher */)
  hydrating = false
  // manually mounted instance, call mounted on self
  // mounted is called for render-created child components in its inserted hook
  if (vm.$vnode == null) {
    vm._isMounted = true
    callHook(vm, 'mounted')
  }
  return vm
}

mountComponent方法定义于core/instance/lifecycle.js中，它接收了三个参数，vm, el, hydrating后两个为可选参数，该函数执行第一步设置vm.$el为el，然后判断有没有正常生成render函数，如果没有$options.render,则通过createEmptyVNode方法创建一个空的VNode给它。接下来检测如果是在开发环境，则判断$options.template存在且第一个字符不是#，或者$options.el存在或el被传入时报一个警告，这是因为我们用了runtime-only版本但是写了template所以命中了这个警告，当然如果不满足上述条件也要报一个template和render函数未定义的警告，因为代码执行到这个环节无论如何也应该有render函数的。执行完这一步之后，调用callHook(vm, 'beforeMount')方法，显而易见这里是调用了生命周期钩子，关于VNode和生命周期我们之后章节讨论。 接着定义了updateComponent，之后就判断环境config.performance 和 mark配置做了一些性能埋点，当项目运行比较卡时可以考虑使用这些性能埋点，这里我们就先不详细分析了。然后我们还是看正常逻辑，赋予updateComponent变量功能为执行vm._update(vm._render(), hydrating)方法，_update方法其实是有两部分逻辑，先是调用vm._render()通过vm.$options.render函数返回VNode，然后通过vm._update方法将VNode渲染成真正的dom。 接着调用了new Watcher()方法，它接收了五个参数，第一个是vm，第二个就是我们上面定义的updateComponent方法，第三个参数noop，实际上就是一个空函数，第四个是一个对象，第五个是判断是否是渲染watcher的标识。 updateComponent方法在new Watcher()时候执行，我们先来分析vm._render()。我们在_init时候调用了initRender方法，_render方法就是在这时候被挂载到Vue原型上的，Vue.prototype._render()方法定义于instance/render.js下

Vue.prototype._render = function (): VNode {
    const vm: Component = this
    const { render, _parentVnode } = vm.$options

    if (_parentVnode) {
      vm.$scopedSlots = normalizeScopedSlots(
        _parentVnode.data.scopedSlots,
        vm.$slots,
        vm.$scopedSlots
      )
    }
  ...
}

可以看到该方法是用于生成VNode的函数，首先取到render函数和_parentVnode，我们接着往下看

    // set parent vnode. this allows render functions to have access
    // to the data on the placeholder node.
    vm.$vnode = _parentVnode
    // render self
    let vnode
    try {
      // There's no need to maintain a stack because all render fns are called
      // separately from one another. Nested component's render fns are called
      // when parent component is patched.
      currentRenderingInstance = vm
      vnode = render.call(vm._renderProxy, vm.$createElement)
    } catch (e) {
      handleError(e, vm, `render`)
      // return error render result,
      // or previous vnode to prevent render error causing blank component
      /* istanbul ignore else */
      if (process.env.NODE_ENV !== 'production' && vm.$options.renderError) {
        try {
          vnode = vm.$options.renderError.call(vm._renderProxy, vm.$createElement, e)
        } catch (e) {
          handleError(e, vm, `renderError`)
          vnode = vm._vnode
        }
      } else {
        vnode = vm._vnode
      }
    } finally {
      currentRenderingInstance = null
    }
    // if the returned array contains only a single node, allow it
    if (Array.isArray(vnode) && vnode.length === 1) {
      vnode = vnode[0]
    }
    // return empty vnode in case the render function errored out
    if (!(vnode instanceof VNode)) {
      if (process.env.NODE_ENV !== 'production' && Array.isArray(vnode)) {
        warn(
          'Multiple root nodes returned from render function. Render function ' +
          'should return a single root node.',
          vm
        )
      }
      vnode = createEmptyVNode()
    }
    // set parent
    vnode.parent = _parentVnode
    return vnode
  }

接下来就执行到render.call(vm._renderProxy，vm.$createElement)方法，之前分析_init时候已经知道vm._renderProxy在生产环境就是vm，在非生产环境是个Proxy的代理，vm.$createElement作为参数传入，注意看这块不是直接调用的，它是作为参数传入的，那vm.$createElement方法又是怎样定义的呢，又是什么时候被调用的呢？我们在分析initRender的时候已经详细分析过了initRender简析，这儿就不在赘述。render方法最终返回的是一个VNode。VNode是什么我们下个小节就会揭晓。如果获取vnode失败在catch中做一些错误处理，最后不管成功失败都把currentRenderingInstance置为null。 再往下我们允许vnode是一个长度为1的数组，这时我们使vnode = vnode[0]。如果vnode不是VNode的一个实例，那么就让vnode等于空的VNode，在这之前判断是在非生产环境且vnode是一个数组的话会报警告说：渲染函数返回了多个根节点，应该返回单个根节点。最后将vnode作为整个vm._render方法的返回值。

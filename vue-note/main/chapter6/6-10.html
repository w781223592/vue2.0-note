# 6.10 slot（中）

  这小节我们先来看具名插槽，首先还是写一个demo

const child = {
  template: `<div>
    我是<slot name="name">1</slot>
  </div>`
}

new Vue({
  el: '#app',
  components: { child },
  template: `<child>
    <div slot="name">子</div>
  </child>`
})

  我们先来看编译时的不同，对于子组件的编译，解析到slot标签时会执行

function processSlotOutlet (el) {
  if (el.tag === 'slot') {
    el.slotName = getBindingAttr(el, 'name')
  }
}

  获取slot标签的name属性并且添加到el.slotName上。对于父组件的编译，遇到带有slot属性的标签会执行

  // slot="xxx"
  const slotTarget = getBindingAttr(el, 'slot')
  if (slotTarget) {
    el.slotTarget = slotTarget === '""' ? '"default"' : slotTarget
    el.slotTargetDynamic = !!(el.attrsMap[':slot'] || el.attrsMap['v-bind:slot'])
  }

  会添加el.slotTarget属性，这里我们是name。然后会执行genData时做一个处理

  // slot target
  // only for non-scoped slots
  if (el.slotTarget && !el.slotScope) {
    data += `slot:${el.slotTarget},`
  }

  在data中添加一个slot: 'name'属性，然后执行resolveSlots时

  if ((child.context === context || child.fnContext === context) &&
    data && data.slot != null
  ) {
    var name = data.slot;
    var slot = (slots[name] || (slots[name] = []));
    if (child.tag === 'template') {
      slot.push.apply(slot, child.children || []);
    } else {
      slot.push(child);
    }
  }

  会执行这一个分支，获取data.slot，也就是我们上面定义的name，然后初始化slots.name为空数组或取现有的slots.name，然后如果child.tag不是template的话就直接将child添加到slot数组中，否则会将child.children添加进去，因为我们知道template是不做渲染的。
  实际上对于普通插槽和具名插槽是没有太大区别的，因为普通插槽其实就是被命名为default的具名插槽。唯一需要处理的就是给其打上不同data.slot名称，这是靠编译时提取slot属性名的做到的。然后生成vnode过程中就可以保存slot名称，然后到这里执行resolveSlots方法其实就是给具有不同slot标识分组添加到vm.$slots。以便于渲染时根据slot的name将其渲染到正确位置。
  真正比较复杂的就是我们接下来的内容，我们知道对于我们插槽它是在父级作用的，但是却是要添加到子组件中，但是如果我们想要在写插槽时用一些子组件中的数据时，Vue也给我们提供了这样的能力，也就是使用作用域插槽。 我们还是写一个demo

const child = {
  template: `<div>
    我是<slot data="子" name="one">1</slot>
  </div>`
}

new Vue({
  el: '#app',
  components: { child },
  template: `<child>
    <div slot="one" slot-scope="scope">{{ scope.data }}</div>
  </child>`
})

  然后我们从编译时的parse阶段说起，对于子组件来说会为解析到的slot标签添加el.attrs.data="子"，el.slotName="one"。对于父组件会执行processSlotContent，解析到slot="one"和slot-scope="scope"然后将el.slotScope设置为"scope"，将el.slotTarget设置为"one"。
  首先我们要回顾一下processSlotContent方法的调用时机，在processElement中会调用processSlotContent和processSlotOutlet方法，其中后者就是我们上面说的对子组件slot添加el.slotName过程。
  而processElement又是在解析到闭合标签时调用closeElement时执行的。接下来又会执行到下面逻辑

  if (currentParent && !element.forbidden) {
    if (element.elseif || element.else) {
      processIfConditions(element, currentParent)
    } else {
      if (element.slotScope) {
        // keep it in the children list so that v-else(-if) conditions can find it as the prev node.
        // 将其保留在子列表中，以便v-else（-if）条件可以将其作为prev节点找到
        const name = element.slotTarget || '"default"'
        ;(currentParent.scopedSlots || (currentParent.scopedSlots = {}))[name] = element
      }
      currentParent.children.push(element)
      element.parent = currentParent
    }
  }

  在执行processSlotContent后就会为带有scope-slot的div标签的el添加一些slotScope属性和slotTarget属性，在demo中其值分别为scope和one。然后接下来是会给父元素添加scopedSlots对象，并给它添加one属性，其值就是我们div元素的element对象。这里的父元素就是child标签生成的element。
  然后到代码生成阶段对于子组件来说genSlot会处理slot标签返回"_t("one", function(){return [_v("1")]},{"data":"子"})"。记住这里我们是有第三个参数的，也就是renderSlot中第三个参数props。
  然后我们看代码生成时对父组件处理，这里就不能单单分析div标签了，还要分析组件标签child，因为我们上面给child元素添加了scopedSlots，那么接下来就来看看在genData时候做了哪些事？

  // scoped slots
  if (el.scopedSlots) {
    data += `${genScopedSlots(el, el.scopedSlots, state)},`
  }

  如果el.scopedSlots存在则会执行genScopedSlots

function genScopedSlots (
  el: ASTElement,
  slots: { [key: string]: ASTElement },
  state: CodegenState
): string {
  // by default scoped slots are considered "stable", this allows child
  // components with only scoped slots to skip forced updates from parent.
  // but in some cases we have to bail-out of this optimization
  // for example if the slot contains dynamic names, has v-if or v-for on them...
  let needsForceUpdate = el.for || Object.keys(slots).some(key => {
    const slot = slots[key]
    return (
      slot.slotTargetDynamic ||
      slot.if ||
      slot.for ||
      containsSlotChild(slot) // is passing down slot from parent which may be dynamic 正在从父级向下传递插槽，这可能是动态的
    )
  })
	// 省略部分代码
  const generatedSlots = Object.keys(slots)
    .map(key => genScopedSlot(slots[key], state))
    .join(',')

  return `scopedSlots:_u([${generatedSlots}]${
    needsForceUpdate ? `,null,true` : ``
  }${
    !needsForceUpdate && needsKey ? `,null,false,${hash(generatedSlots)}` : ``
  })`
}

  这个方法定义比较复杂，中间有一些针对bug的处理。这里slots就是我们传入的el.scopedSlots。然后看这一段注释，大致意思是说在默认情况下，作用域插槽被认为是稳定的，这允许只有作用域插槽的子组件跳过来自父组件的强制更新。但是一些特殊情况需要处理，比如插槽含有动态名称，或者有v-if或者v-for的情况。
  这里判断是否需要强制更新子组件，如果el是v-for渲染的就需要强制更新。否则判断其scopedSlots中的元素是否是动态插槽名称，或者使用v-if或者使用v-for，或者满足containsSlotChild方法校验时就需要强制更新。

function containsSlotChild (el: ASTNode): boolean {
  if (el.type === 1) {
    if (el.tag === 'slot') {
      return true
    }
    return el.children.some(containsSlotChild)
  }
  return false
}

  判断子元素中是否又有slot标签，如果有就返回true，否则为false。
  然后执行genScopedSlot方法

function genScopedSlot (
  el: ASTElement,
  state: CodegenState
): string {
  const isLegacySyntax = el.attrsMap['slot-scope']
  if (el.if && !el.ifProcessed && !isLegacySyntax) {
    return genIf(el, state, genScopedSlot, `null`)
  }
  if (el.for && !el.forProcessed) {
    return genFor(el, state, genScopedSlot)
  }
  const slotScope = el.slotScope === emptySlotScopeToken
    ? ``
    : String(el.slotScope)
  const fn = `function(${slotScope}){` +
    `return ${el.tag === 'template'
      ? el.if && isLegacySyntax
        ? `(${el.if})?${genChildren(el, state) || 'undefined'}:undefined`
        : genChildren(el, state) || 'undefined'
      : genElement(el, state)
    }}`
  // reverse proxy v-slot without scope on this.$slots
  const reverseProxy = slotScope ? `` : `,proxy:true`
  return `{key:${el.slotTarget || `"default"`},fn:${fn}${reverseProxy}}`
}

  这里传入的el是slots中取到的，也就是说这里el是我们div而不children。所以这里会取到slot-scope属性scope放到isLegacySyntax。这个变量译为是否是传统语法，关于这个传统语法问题我们后面会说明。
  这里如果el.if存在并且没有被处理过的话，那么会先调用genIf，然后会传入它本身作为第三个参数，也就是genScopedSlot。这样有什么用呢？我们来看genIf

export function genIf (
  el: any,
  state: CodegenState,
  altGen?: Function,
  altEmpty?: string
): string {
  el.ifProcessed = true // avoid recursion
  return genIfConditions(el.ifConditions.slice(), state, altGen, altEmpty)
}

  就是作为我们之前没说过的altGen，这里会再次传入genIfConditions

function genIfConditions (
  conditions: ASTIfConditions,
  state: CodegenState,
  altGen?: Function,
  altEmpty?: string
): string {
  if (!conditions.length) {
    return altEmpty || '_e()'
  }

  const condition = conditions.shift()
  if (condition.exp) {
    return `(${condition.exp})?${
      genTernaryExp(condition.block)
    }:${
      genIfConditions(conditions, state, altGen, altEmpty)
    }`
  } else {
    return `${genTernaryExp(condition.block)}`
  }

  // v-if with v-once should generate code like (a)?_m(0):_m(1)
  function genTernaryExp (el) {
    return altGen
      ? altGen(el, state)
      : el.once
        ? genOnce(el, state)
        : genElement(el, state)
  }
}

  然后genIfConditions处理if相关逻辑就时会回调altGen然后来完成这部分逻辑的处理，回到genScopedSlot方法中，如果绑定了v-for指令也是差不多的处理。
  回到genScopedSlot中，如果el.slotScope等于emptySlotScopeToken那就使slotScope为空

export const emptySlotScopeToken = `_empty_`

  这个东西我们后面再去说明。这里我们el.slotScope为scope，所以slotScope就被赋值为'scope'。
  然后会定义·生成一个函数代码串，函数return内容我们简单翻译一下：
  如果tag为template那么使用genElement方法返回代码。
  如果不是template：
    如果el.if为true并且为传统语法，返回一个三元表达式的字符串
    否则返回genChildren的执行结果或者undefined
  返回的三元表达判断el.if为true那么返回genChildren的执行结果，不为true则返回undefined。因为我们template不会渲染到页面，所以直接genChildren去处理子节点即可，而不是去使用genElement。
  reverseProxy则是返回一个空字符串，如果不是传统语法会返回,proxy:true 最后返回一个：
"{key:"one",fn:function(scope){return _c('div',{},[_v(_s(scope.data))])}}"
  这样的结果。回到genScopedSlots方法中，generatedSlots就是这样一个结果。
  其实在这之前genScopedSlots方法还会有些复杂的处理

  // 如果一个具有作用域插槽的组件位于条件分支中，那么可以重用相同的组件，但是编译后的插槽内容可能不同。为了避免这种情况，我们可以根据生成的所有插槽内容的代码生成一个唯一的键
  let needsKey = !!el.if
  // 或者当它在另一个作用域槽或v-for内时（响应性可能由于中间作用域变量而断开）
  // 通过正确分析作用域内绑定并跳过实际不使用作用域变量的强制更新，可以进一步优化这一点。
  if (!needsForceUpdate) {
    let parent = el.parent
    while (parent) {
      if (
        (parent.slotScope && parent.slotScope !== emptySlotScopeToken) ||
        parent.for
      ) {
        needsForceUpdate = true
        break
      }
      if (parent.if) {
        needsKey = true
      }
      parent = parent.parent
    }
  }

  这里说一个具有作用域插槽组件在条件分支中，可以重用相同的组件，但是编译后的插槽内容可能不同。为了解决这个问题，如果el.if为true那么会使needsKey为true，我们会为插槽内容生成一个唯一的key来区别。
  然后接下来如果needsForceUpdate为false，那就去向上寻找父元素，如果父元素也在一个作用域插槽中或者在v-for中会将needsForceUpdate置为true。如果父元素有被v-if控制那么needsKey也很会置为true。
最后一段代码

  return `scopedSlots:_u([${generatedSlots}]${
    needsForceUpdate ? `,null,true` : ``
  }${
    !needsForceUpdate && needsKey ? `,null,false,${hash(generatedSlots)}` : ``
  })`

  如果needsForceUpdate为true则后面会有null,true两个参数，如果为false但是needsKey为true，则后面有null,false,和hash(generatedSlots)三个参数。
  hash就是一个特殊的运算处理

function hash(str) {
  let hash = 5381
  let i = str.length
  while(i) {
    hash = (hash * 33) ^ str.charCodeAt(--i)
  }
  return hash >>> 0
}

  就是去生成一个唯一的key值。
  最后返回就是这样一个结果"scopedSlots:_u([{key:"one",fn:function(scope){return _c('div',{},[_v(_s(scope.data))])}}]),"。然后就会把这个东西给添加到data中。也就是说child的data中会添加scopedSlots属性。
我们最后生成的代码是： _c('child',{scopedSlots:_u([{key:"one",fn:function(scope){return _c('div',{},[_v(_s(scope.data))])}}])})，那么在生成vnode时，组件child的vnode就会有这个属性了，这个需要我们注意一下，我们在下个小节分析运行时的实现就会用到它。

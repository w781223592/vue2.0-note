# 6.1 events（4）

  那么这个小节我们就来分析组件自定义事件，我们来回顾一下组件初始化的过程，经过编译后我们的模板全部转化为render函数，在调用render函数过程中会使用h函数处理组件，h也就是我们createElement函数，然后又会执行_createElement函数，然后解析到传入的tag是个string并且不是平台保留标签，那么就是个组件标签，对于组件标签会执行上小节提到的createComponent方法去生成组件vnode，这时将data.on保存到listeners中，然后执行

  const vnode = new VNode(
    `vue-component-${Ctor.cid}${name ? `-${name}` : ''}`,
    data, undefined, undefined, undefined, context,
    { Ctor, propsData, listeners, tag, children },
    asyncFactory
  )

  创建vnode后作为一个配置项传入。然后在调用vm_update方法进入patch逻辑，在patch过程中支持createElm方法会调用createComponent，同样是这个名字但是不同的方法。
  这时会执行vnode.data.hook.init方法，然后执行

 const child = vnode.componentInstance = createComponentInstanceForVnode(
    vnode,
    activeInstance
  )

createComponentInstanceForVnode，这个方法

  const options: InternalComponentOptions = {
    _isComponent: true, // 记住这个标识，在Vue初始化逻辑中就对它进行了判断
    _parentVnode: vnode,
    parent
  }
  return new vnode.componentOptions.Ctor(options)

  会用new创建新的Vue实例，也就是回到Vue初始化的逻辑中了

  if (options && options._isComponent) {
    initInternalComponent(vm, options)
  }

  如果是一个组件就用initInternalComponent合并options

export function initInternalComponent(vm: Component, options: InternalComponentOptions) {
  const opts = vm.$options = Object.create(vm.constructor.options)
  // doing this because it's faster than dynamic enumeration.
  // 这样做是因为它比动态枚举更快
  const parentVnode = options._parentVnode
  opts.parent = options.parent
  opts._parentVnode = parentVnode

  const vnodeComponentOptions = parentVnode.componentOptions
  opts.propsData = vnodeComponentOptions.propsData
  opts._parentListeners = vnodeComponentOptions.listeners
  opts._renderChildren = vnodeComponentOptions.children
  opts._componentTag = vnodeComponentOptions.tag

  if (options.render) {
    opts.render = options.render
    opts.staticRenderFns = options.staticRenderFns
  }
}

  这里vnodeComponentOptions对象就是我们创建vnode时传入的配置项

{ Ctor, propsData, listeners, tag, children }

  这里就有listeners，然后init函数后面会初始化事件initEvents，这部分逻辑是我们之前没有分析过的，我们现在就来分析它。
initEvents定义于core/instance/events.js中

export function initEvents (vm: Component) {
  vm._events = Object.create(null)
  vm._hasHookEvent = false
  // init parent attached events
  const listeners = vm.$options._parentListeners
  if (listeners) {
    updateComponentListeners(vm, listeners)
  }
}

  先将vm._events初始化为一个空对象，将vm._hasHookEvent初始化为false。然后获取到我们listeners对象，如果不为空的话执行updateComponentListeners

export function updateComponentListeners (
  vm: Component,
  listeners: Object,
  oldListeners: ?Object
) {
  target = vm
  updateListeners(listeners, oldListeners || {}, add, remove, createOnceHandler, vm)
  target = undefined
}

  就是调用updateListeners，但是add、remove、createOnceHandler函数不同。因为我们updateListeners首先执行了createOnceHandler，也就是处理once事件，所以我们先来看它

function createOnceHandler (event, fn) {
  const _target = target
  return function onceHandler () {
    const res = fn.apply(null, arguments)
    if (res !== null) {
      _target.$off(event, onceHandler)
    }
  }
}

  其实就和之前dom事件一样，只不过之前target是真实dom，而这里是vm实例，之前会调用remove方法，而这里调用的是vm.$off。
  然后我们看add方法

function add (event, fn) {
  target.$on(event, fn)
}

  这也很简单就是调用vm.$on。我们来看$on方法是怎么实现的，在当前文件中定义有eventsMixin方法，我们知道mixin系列方法会往Vue原型添加一些内容。

  const hookRE = /^hook:/
  Vue.prototype.$on = function (event: string | Array<string>, fn: Function): Component {
    const vm: Component = this
    if (Array.isArray(event)) {
      for (let i = 0, l = event.length; i < l; i++) {
        vm.$on(event[i], fn)
      }
    } else {
      (vm._events[event] || (vm._events[event] = [])).push(fn)
      // optimize hook:event cost by using a boolean flag marked at registration
      // instead of a hash lookup
      if (hookRE.test(event)) {
        vm._hasHookEvent = true
      }
    }
    return vm
  }

  如果是event是一个数组的话就循环调用vm.$on方法。不是数组的话就将函数添加到vm._events[event]数组中。如果是监听hook，那么就将vm._hasHookEvent置为true。这个是用来监听声明周期的，比如定义@hook:created事件。
  然后我们来看remove事件

function remove (event, fn) {
  target.$off(event, fn)
}

  也就是vm.$off函数

  Vue.prototype.$off = function (event?: string | Array<string>, fn?: Function): Component {
    const vm: Component = this
    // all
    if (!arguments.length) {
      vm._events = Object.create(null)
      return vm
    }
    // array of events
    if (Array.isArray(event)) {
      for (let i = 0, l = event.length; i < l; i++) {
        vm.$off(event[i], fn)
      }
      return vm
    }
    // specific event
    const cbs = vm._events[event]
    if (!cbs) {
      return vm
    }
    if (!fn) {
      vm._events[event] = null
      return vm
    }
    // specific handler
    let cb
    let i = cbs.length
    while (i--) {
      cb = cbs[i]
      if (cb === fn || cb.fn === fn) {
        cbs.splice(i, 1)
        break
      }
    }
    return vm
  }

  这个事件稍微有些复杂，因为对于dom事件最后都通过一个invoker事件来管理，所以移除也是比较简单的。但是$on则是放到一个数组中来管理，因为绑定到模板的事件是可以统一管理的，但是不排除我们手动调用vm方法去添加和删除事件。
  首先这里参数都是可选的，如果不传参数的话说明是要卸载所有事件，也就是我们arguments长度为0执行的逻辑。
  如果event是一个数组的话，同样需要遍历调用vm.$off去卸载事件。如果不是数组那可能是特定事件名，然后获取到该事件，如果不存在该事件的话就什么都不做返回。
  接下来判断如果没有传入fn，那就将相同事件名整个移除，如果传入fn则要在cbs中找到对应事件函数删除。这里的cb.fn是什么我们先不去分析。
  我们先看vm.$once方法

  Vue.prototype.$once = function (event: string, fn: Function): Component {
    const vm: Component = this
    function on () {
      vm.$off(event, on)
      fn.apply(vm, arguments)
    }
    on.fn = fn
    vm.$on(event, on)
    return vm
  }

  就是定义一个执行一次就会销毁的函数并且将它挂在on.fn上，这样我们就可以手动去销毁它了，因为我们外部无法访问到on。这也就是上面为什么要判断cb.fn === fn的原因了。
  接下来我们来看事件的调用方法，也就是$emit方法

  Vue.prototype.$emit = function (event: string): Component {
    const vm: Component = this
    if (process.env.NODE_ENV !== 'production') {
      const lowerCaseEvent = event.toLowerCase()
      if (lowerCaseEvent !== event && vm._events[lowerCaseEvent]) {
        tip(
          `Event "${lowerCaseEvent}" is emitted in component ` +
          `${formatComponentName(vm)} but the handler is registered for "${event}". ` +
          `Note that HTML attributes are case-insensitive and you cannot use ` + `v-on to listen to camelCase events when using in-DOM templates. ` + `You should probably use "${hyphenate(event)}" instead of "${event}".`
        )
      }
    }
    let cbs = vm._events[event]
    if (cbs) {
      cbs = cbs.length > 1 ? toArray(cbs) : cbs
      const args = toArray(arguments, 1)
      const info = `event handler for "${event}"`
      for (let i = 0, l = cbs.length; i < l; i++) {
        invokeWithErrorHandling(cbs[i], vm, args, vm, info)
      }
    }
    return vm
  }

  如果我们事件使用驼峰命名，并且命名转化为小写字母后能在vm._events找到该命名的事件。也就是说如果父组件中监听的是小写形式，但是触发了驼峰形式的事件。那么要提示说：事件xxx在组件xxx中触发，但其非注册为小写形式，注意，HTML属性不区分大小写，在DOM模板中使用时不能使用v-on来侦听camelCase事件。您可能应该使用“连字符事件”而不是驼峰。
  然后取event事件放到cbs中，如果cbs存在，那需要判断cbs是否是一个类数组，如果为类数组那么要转化为真实数组args。
  然后将调用时传入的参数取出来放到一个数组中，然后遍历cbs数组调用所有的回调函数并且会将args参数列表传入。
  到这里我们事件的全部内容就分析完成了。对于生命周期的监听事件，我们在3.4小节开头就可以看到，虽然我们没有去分析，但它是很好理解的。那么下一个小节我们来看v-model的实现。

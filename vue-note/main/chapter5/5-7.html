# 5.7 parse之生成AST（3）

  上小节我们分析到闭合标签的方法closeElement，这小节我们就来详细分析它，先来看它源码： 首先调用了这样一个方法

  function trimEndingWhitespace (el) {
    // remove trailing whitespace node
    if (!inPre) {
      let lastNode
      while (
        (lastNode = el.children[el.children.length - 1]) &&
        lastNode.type === 3 &&
        lastNode.text === ' '
      ) {
        el.children.pop()
      }
    }
  }

  如果不是在inPre标签中，如果el.children从末尾开始有文本节点，并且文本节点内容为空那就删除它，这段代码主要是用来移除标签尾部空节点的，当然在pre标签中则不执行，因为我们知道pre标签会保留文本空白格式。
  接下来执行

  if (!inVPre && !element.processed) {
    element = processElement(element, options)
  }

  满足!inVPre && !element.processed，满足条件执行processElement方法，又是一个process系列方法，但是这个方法定义感觉有些复杂，因为其它命名都是一个单独的小模块定义，而它则是定义为处理element。

export function processElement (
  element: ASTElement,
  options: CompilerOptions
) {
  processKey(element)

  // determine whether this is a plain element after
  // removing structural attributes
  element.plain = (
    !element.key &&
    !element.scopedSlots &&
    !element.attrsList.length
  )

  processRef(element)
  processSlotContent(element)
  processSlotOutlet(element)
  processComponent(element)
  for (let i = 0; i < transforms.length; i++) {
    element = transforms[i](element, options) || element
  }
  processAttrs(element)
  return element
}

  果然它又调用了许多process系列方法，是一个复合方法。我们一步一步往下看，首先是processKey方法

function processKey (el) {
  const exp = getBindingAttr(el, 'key')
  if (exp) {
    if (process.env.NODE_ENV !== 'production') {
      if (el.tag === 'template') {
        warn(
          `<template> cannot be keyed. Place the key on real elements instead.`,
          getRawBindingAttr(el, 'key')
        )
      }
      if (el.for) {
        const iterator = el.iterator2 || el.iterator1
        const parent = el.parent
        if (iterator && iterator === exp && parent && parent.tag === 'transition-group') {
          warn(
            `Do not use v-for index as key on <transition-group> children, ` +
            `this is the same as not using keys.`,
            getRawBindingAttr(el, 'key'),
            true /* tip */
          )
        }
      }
    }
    el.key = exp
  }
}

  看似比较复杂，其实就是做了一件事，取元素的key值将其赋值给el.key。其它逻辑都是为了给我开发环境报出错误提示。
  如果我们给一个template绑定了key，它会报一个警告，当然这仅限于vue2。然后就是对v-for指令的key做校验，当v-for循环时绑定了index作为key值的话，并且其父元素是一个transition-group组件时，那么就报一个警告，transition-group是vue内置组件，我在后面的章节会分析。如果使用index作为key值就相当于没有设置key，这在一段不会修改的循环中是可行的，但是如果会发生变化的话等同于没有key值，这在使用transition-group时显然是有问题的。
  接下来看

  // determine whether this is a plain element after
  // removing structural attributes
  // 在删除结构属性后确定这是否是普通元素
  element.plain = (
    !element.key &&
    !element.scopedSlots &&
    !element.attrsList.length
  )

  如果没有key值、作用域插槽、属性，那就认为它是一个纯的元素，打上一个plain标记。这里说在删除结构属性后，这是什么意思呢？我们知道getAndRemoveAttr可以用来获取并删除属性，那么现在仔细想想我们删除了什么属性呢？v-pre、v-for、v-if等，这些都是指令，换句话说这都是不需要的，既不能作为dom属性也不能作为props，所以我们把它删掉了。v-for、v-i、v-once等指令其实就被叫做结构属性，注意这里没有说v-pre。这些指令会影响结构变化指令并称为结构指令，这在start函数中就有体现。
  接下来是执行四个process系列函数，我们先看processRef

function processRef (el) {
  const ref = getBindingAttr(el, 'ref')
  if (ref) {
    el.ref = ref
    el.refInFor = checkInFor(el)
  }
}

  获取ref值，并将el.ref设置为ref，将el.refInFor设置为checkInFor(el)返回结果

function checkInFor (el: ASTElement): boolean {
  let parent = el
  while (parent) {
    if (parent.for !== undefined) {
      return true
    }
    parent = parent.parent
  }
  return false
}

  checkInFor就是检查是否是在for循环中，先检查自身for属性是否存在，不存在则一直往父级查找直到找到parent.for存在就返回true，否则返回false。
  接下来是processSlotContent方法，插槽内容比较多，我们可以简单了解一下，后面等到了指令章节我们再来专门分析插槽内容。
  先来看两个错误提示

  if (el.tag === 'template') {
    slotScope = getAndRemoveAttr(el, 'scope')
    /* istanbul ignore if */
    if (process.env.NODE_ENV !== 'production' && slotScope) {
      warn(
        `the "scope" attribute for scoped slots have been deprecated and ` +
        `replaced by "slot-scope" since 2.5. The new "slot-scope" attribute ` +
        `can also be used on plain elements in addition to <template> to ` +
        `denote scoped slots.`,
        el.rawAttrsMap['scope'],
        true
      )
    }
    el.slotScope = slotScope || getAndRemoveAttr(el, 'slot-scope')
  } else if ((slotScope = getAndRemoveAttr(el, 'slot-scope'))) {
    /* istanbul ignore if */
    if (process.env.NODE_ENV !== 'production' && el.attrsMap['v-for']) {
      warn(
        `Ambiguous combined usage of slot-scope and v-for on <${el.tag}> ` +
        `(v-for takes higher priority). Use a wrapper <template> for the ` +
        `scoped slot to make it clearer.`,
        el.rawAttrsMap['slot-scope'],
        true
      )
    }
    el.slotScope = slotScope
  }

  第一个报错比较容易理解，之前是通过scope属性定义作用域插槽，在v2.5以后则是使用slot-scope属性，虽然报了警告但是还仍然兼容scope。我们作用域插槽scope是要写在template标签上的，所以我们先要判断template标签才会获取scope，这里也指出slot-scope的好处就是不受template限制。
  第二个报错就是针对不受template限制的slot-scope，如果slot-scope与v-for一起使用，由于v-for优先级更高，就会出现问题。所以好的做法是将slot-scope写在template上，比如:

    <div slot-scope="scope" v-for="item of scope.list"></div>
    <!-- 应该写为： -->
    <template slot-scope="scope">
      <div v-for="item of scope.list"></div>
    </template>

  接下来是对插槽处理

  // slot="xxx"
  const slotTarget = getBindingAttr(el, 'slot')
  if (slotTarget) {
    el.slotTarget = slotTarget === '""' ? '"default"' : slotTarget
    el.slotTargetDynamic = !!(el.attrsMap[':slot'] || el.attrsMap['v-bind:slot'])
    // preserve slot as an attribute for native shadow DOM compat
    // only for non-scoped slots.
    if (el.tag !== 'template' && !el.slotScope) {
      addAttr(el, 'slot', slotTarget, getRawBindingAttr(el, 'slot'))
    }
  }

  也就是对具名插槽处理，先获取插槽名称，这里就不得不提到getBindingAttr的一个巧妙机制了，在遇到静态属性时会做一次JSON字符串的转换，所以即使我们定义slot=''，也会被转换为'""'，从而进入slotTarget判断，然后将el.slotTarget定义为'"default"'或者slotTarget。然后会根据attrsMap中属性判断是否是动态的绑定的，如果是的话就将el.slotTargetDynamic置为true。
  再往下则是用来保存原生影子DOM（shadow DOM）的slot属性，那么这个时候要保证tag不能是template，并且不能是作用域插槽。关于原生影子DOM，我们可以查看如下连接： 影子DOM (opens new window)
  最后如果是原生slot属性则要通过addAttr方法添加到el.attrs中。
  再往下则是v2.6+的v-slot，这个我们就不在这里细说了。
  下一个processSlotOutlet是用来处理slot标签的。

// handle <slot/> outlets
function processSlotOutlet (el) {
  if (el.tag === 'slot') {
    el.slotName = getBindingAttr(el, 'name')
    if (process.env.NODE_ENV !== 'production' && el.key) {
      warn(
        `\`key\` does not work on <slot> because slots are abstract outlets ` +
        `and can possibly expand into multiple elements. ` +
        `Use the key on a wrapping element instead.`,
        getRawBindingAttr(el, 'key')
      )
    }
  }
}

  对于具名插槽会获取其name值并保存在el.slotName中。接下来是slot标签是否设置key值做一个检验，因为slot中可能对应多个节点，所以设置key是不会工作的。
  接下来我们来看processComponent方法

function processComponent (el) {
  let binding
  if ((binding = getBindingAttr(el, 'is'))) {
    el.component = binding
  }
  if (getAndRemoveAttr(el, 'inline-template') != null) {
    el.inlineTemplate = true
  }
}

  对is属性处理，会将el.component设置为其值，如果设置inline-template则将el.inlineTemolate设置为true。这是内联模板的使用，因为用的是在不多，就不做过多介绍了。
  接着往下看processElement逻辑

  for (let i = 0; i < transforms.length; i++) {
    element = transforms[i](element, options) || element
  }
  processAttrs(element)
  return element

  就是循环调用transform中的方法，这同样是我们modules中的方法，我们还是放到compile文档中去分析。
  最后执行processAttrs方法后将element返回。processAttrs方法非常庞大，因为它会用来处理所有剩余属性。
  我们先来总结一下前面分析了的属性：
  指令：v-pre、v-for、v-if、v-else-if、v-else、v-once
  属性：key、ref、slot、slot-scope、scope、name、is、inline-template
  我们处理这些属性的方法都是调动getAndRemoveAttr方法去处理，也就是取到数据后会给el添加相应的标识，之后就会把其从属性列表中删除。但是并不是所有地方都是直接使用getAndRemoveAttr，而是通过getBindingAttr方法去调用。那么直接调用和非直接调用有什么区别呢？
  getAndRemoveAttr会根据属性名去获取属性值并返回，而getBindingAttr则是做了一层封装，会优先获取动态属性，找不到动态属性才会尝试获取静态属性。但是静态属性的获取才是最大区别，使用getBindingAttr处理静态属性会通过JSON.stringify方法额外处理。
  对于我们v-if绑定表达式如果解析到'a || b'字符串，最后通过new Function处理会变成a || b表达式。如果用getBindingAttr处理的话会变成'"a || b"'。最后同通过new Function会处理为字符串'a || b'，显然这是不对的。而假如我们slot传入name="head"，如果通过getAndRemoveAttr处理的话最后生成代码就变为了变量head。这显然也是有问题的。所以这里有两种方法去处理属性。
  其实这里也可以看到，getBindingAttr方法主要是处理一些既可以传动态值又能传静态值的属性。像之前我们分析的这些指令我们都是绑定动态表达式或变量，所以它们都是getAndRemoveAttr来处理，而属性大部分都是既可以传静态值也可以动态绑定，所以大部分用getBindingAttr处理的，除了作用域插槽和内联模板。

# 5.6 parse之生成AST（2）

  接着上一小节我们继续来分析start钩子函数

  if (!inVPre) {
    processPre(element)
    if (element.pre) {
      inVPre = true
    }
  }

  可以看到这段代码执行了一个processPre方法，在parse方法中有很多process开头的方法，它们是用来处理特定标签的，它会在不同元素的AST上添加各自独特的属性来使生成的AST对元素的描述更加丰满。
  这里在非inPre的情况下会执行if内的逻辑，我们知道inVPre一开始是被初始化为false的，所以这部分逻辑肯定会被执行。
  我们来看processPre方法

function processPre (el) {
  if (getAndRemoveAttr(el, 'v-pre') != null) {
    el.pre = true
  }
}

  就是去判断有无v-pre指令，对于一些静态的内容可以添加v-pre指令跳过这个元素与它子元素的编译过程，使编译加快。解析到v-pre标签后它会给el添加pre属性为true。
  然后接下来判断element.pre为true话就将inVPre变量设置为true。
  接下来

  if (platformIsPreTag(element.tag)) {
    inPre = true
  }

  判断是否是pre标签，如果是的话把inPre设置为true

  if (inVPre) {
    processRawAttrs(element)
  } else if (!element.processed) {
    // structural directives
    processFor(element)
    processIf(element)
    processOnce(element)
  }

  接下来判断inVPre为true的话执行processRawAttrs方法

function processRawAttrs (el) {
  const list = el.attrsList
  const len = list.length
  if (len) {
    const attrs: Array<ASTAttr> = el.attrs = new Array(len)
    for (let i = 0; i < len; i++) {
      attrs[i] = {
        name: list[i].name,
        value: JSON.stringify(list[i].value)
      }
      if (list[i].start != null) {
        attrs[i].start = list[i].start
        attrs[i].end = list[i].end
      }
    }
  } else if (!el.pre) {
    // non root node in pre blocks with no attributes
    el.plain = true
  }
}

  获取到attrsList数组并获取长度，如果长度不为0就创建一个相同长度的数组并且把它赋值给el.attrs，然后遍历attrs处理为和attrsList一样的数组，但是这里对其元素的value值做了JSON.stringify转化，原因和之前一样，就不再阐述了。如果元素没有属性，并且没有v-pre指令就给el打上一个plain标记，表示元素是纯的。
  如果inVPre为false，并且!element.processed为true，element.processed是我们在之前执行preTransforms中方法时增加的属性，接下来我们看下面的几个process系列方法

export function processFor (el: ASTElement) {
  let exp
  if ((exp = getAndRemoveAttr(el, 'v-for'))) {
    const res = parseFor(exp)
    if (res) {
      extend(el, res)
    } else if (process.env.NODE_ENV !== 'production') {
      warn(
        `Invalid v-for expression: ${exp}`,
        el.rawAttrsMap['v-for']
      )
    }
  }
}

  定义exp去获取v-for的属性，如果exp有值的话就通过parseFor方法去处理生成res，如果res存在的话就把res中的内容合并到el，否则报一个警告说无效的v-for表达式。
  下面来看parseFor方法

type ForParseResult = {
  for: string;
  alias: string;
  iterator1?: string;
  iterator2?: string;
};

export function parseFor (exp: string): ?ForParseResult {
  const inMatch = exp.match(forAliasRE)
  if (!inMatch) return
  const res = {}
  res.for = inMatch[2].trim()
  const alias = inMatch[1].trim().replace(stripParensRE, '')
  const iteratorMatch = alias.match(forIteratorRE)
  if (iteratorMatch) {
    res.alias = alias.replace(forIteratorRE, '').trim()
    res.iterator1 = iteratorMatch[1].trim()
    if (iteratorMatch[2]) {
      res.iterator2 = iteratorMatch[2].trim()
    }
  } else {
    res.alias = alias
  }
  return res
}

  根据类型定义可以看到最后返回的是一个对象，包括for、alias两个必有属性，还有interator1和interator2两个非必须属性。
  这个方法看似复杂实则非常简单，就是通过forAliasRE正则去match传入的exp字符串，exp其实就是'item in list'这种形式，如果没有匹配到结果就直接什么都不做。
  如果匹配到了就定义res为一个空对象，这里为了方便理解我们先看看inMatch是怎样一个结果

const inMatch = [
  'item in list',
  'item',
  'list'
]

  这里其实是两个捕获组，一个捕获遍历的元素定义——item，一个捕获遍历的数组——list。所以这里给res.for赋值为list，也就是遍历的是list数组，接着要去处理第一个捕获组的item。因为我们还可能去获取遍历的索引，这个时候可能是(item,index)这种形式，并且前后还可能带有空格，所以我们先去掉首位空格再通过stripParensRE正则将括号去掉，这样就得到了'item, index'，然后把这个结果保存在alias中。
  接下来通过forIteratorRE去匹配alias，匹配结果保存在iteratorMatch中，如果没有匹配到说明是上面只有item的情况，那么直接把alias赋值给res.alias。如果匹配到了，那么就将alias只截取前面的item保存到res.alias中。然后第二个值index会在iteratorMatch第一个捕获组中，把它赋值给res.iterator1。同样的我们之前说过如果是遍历对象可能是'value, key, index'形式，可能就会有第三个值了，这时把它放到res.iterator2中。这就是对v-for的处理。
  再往下看processIf，显然我们可以猜到它是对v-if的处理，下面我们来看看它做了怎样的处理

function processIf (el) {
  const exp = getAndRemoveAttr(el, 'v-if')
  if (exp) {
    el.if = exp
    addIfCondition(el, {
      exp: exp,
      block: el
    })
  } else {
    if (getAndRemoveAttr(el, 'v-else') != null) {
      el.else = true
    }
    const elseif = getAndRemoveAttr(el, 'v-else-if')
    if (elseif) {
      el.elseif = elseif
    }
  }
}

  同样定义了exp获取v-if的属性值，然后将el.if赋值为exp，然后执行addIfCondition方法。如果没有获取到v-if的属性值，那就要在else逻辑中判断有没有v-else或者v-else-if指令。这里是判断v-else的属性值!= null，因为我们在写v-else是不会写属性值的，那它就会被处理为''，只要我们添加了某个属性而没有赋值都会默认处理为''，这个在handleStartTag方法中就有体现。所以这里就要判断!= null，而不是和v-if一样的判断，这里对v-else的处理是将el.else置为true。然后还要判断v-else-if属性，如果存在的话将el.elseif赋值为elseif。
  然后我们回过头来看addIfCondition方法

export function addIfCondition (el: ASTElement, condition: ASTIfCondition) {
  if (!el.ifConditions) {
    el.ifConditions = []
  }
  el.ifConditions.push(condition)
}

  如果没有el.ifConditions就将其初始化为空数组，然后将第二个参数condition添加到数组中。
  接下来是processOnce，也就是对v-once的处理

function processOnce (el) {
  const once = getAndRemoveAttr(el, 'v-once')
  if (once != null) {
    el.once = true
  }
}

  同样的过程，如果有v-once指令将el.once设置为true。
  接着往下看start钩子：

  if (!root) {
    root = element
    if (process.env.NODE_ENV !== 'production') {
      checkRootConstraints(root)
    }
  }

  如果非root则执行其中逻辑，root含义是根元素，在第一次解析到开始标签是它肯定是没有值的，所以这里直接把root赋值为element就行了。然后还要对它进行了一个检测：

  function checkRootConstraints (el) {
    if (el.tag === 'slot' || el.tag === 'template') {
      warnOnce(
        `Cannot use <${el.tag}> as component root element because it may ` +
        'contain multiple nodes.',
        { start: el.start }
      )
    }
    if (el.attrsMap.hasOwnProperty('v-for')) {
      warnOnce(
        'Cannot use v-for on stateful component root element because ' +
        'it renders multiple elements.',
        el.rawAttrsMap['v-for']
      )
    }
  }

  我们都知道在vue2组件模板中作为一个根元素其实是有限制的，就是只允许出现单个根标签，也就是第一个解析到的标签不能有兄弟元素。
  所以这里就是基于这个限制来判断的，当然这里不是判断限制条件的地方，而是先把可能出现多个根元素的情况剔除，比如是一个slot插槽，或者一个template元素。对于slot很好理解，你不能确定用户在插槽里放置几个根节点，并且也并没有限制插槽中药使用单个根节点，slot和template属于不会真正渲染到dom的标签，所以真正渲染其字内容的话要根据子节点的根节点数量来考虑，针对这种不确定因数就要做一个限制。同样对使用v-for指令的元素也是一个道理。
  继续往下看start钩子

  if (!unary) {
    currentParent = element
    stack.push(element)
  } else {
    closeElement(element)
  }

  如果不是一元标签就将currentParent设置为当前元素，因为这里我们解析到开始标签才执行start钩子，对于双标签来说，接下来解析的将会是其子内容，如果再往后解析到开始标签那么上一个开始标签就是它的父元素，所以这里把currentParent赋值为当前元素，就是为下一个再解析到开始标签准备的。
  接下来再将element添加到stack数组中，我们在parseHTML中也定义了一个stack，是用来存放“打开的标签”的，那么这里它也应该是相同的作用。
  如果是一个一元标签的话，我们就应该调用closeElement方法将元素标签“关闭”。到这里我们的start钩子就执行完了，那我们接下来看看如果解析到关闭标签时执行的end钩子做了什么事，因为它看起来十分简单。

  end (tag, start, end) {
    const element = stack[stack.length - 1]
    // pop stack
    stack.length -= 1
    currentParent = stack[stack.length - 1]
    if (process.env.NODE_ENV !== 'production' && options.outputSourceRange) {
      element.end = end
    }
    closeElement(element)
  },

  首先获取stack栈顶元素，然后使stack的length – 1。然后把currentParent恢复到上一级。然后满足条件就给element.end赋值，这个没什么好说的了。
  最后其实才是end方法的重点内容，我们在上面start钩子中遇到一元标签也会执行这个方法的，这个方法比较复杂，我们把它放到下一个小节去分析。

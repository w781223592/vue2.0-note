# 5.10 optimize优化

  前面小节我们分析了生成AST的parse函数，接下来我们继续分析baseCompile函数的第二个函数optimize优化过程

  const ast = parse(template.trim(), options)
  if (options.optimize !== false) {
    optimize(ast, options)
  }
  const code = generate(ast, options)

optimize函数定义于compiler/optimizer.js中，不同于parse方法，optimize方法是非常简短的，我们来看源码

/**
 * Goal of the optimizer: walk the generated template AST tree
 * and detect sub-trees that are purely static, i.e. parts of
 * the DOM that never needs to change.
 *
 * Once we detect these sub-trees, we can:
 *
 * 1. Hoist them into constants, so that we no longer need to
 *    create fresh nodes for them on each re-render;
 * 2. Completely skip them in the patching process.
 */
export function optimize (root: ?ASTElement, options: CompilerOptions) {
  if (!root) return
  isStaticKey = genStaticKeysCached(options.staticKeys || '')
  isPlatformReservedTag = options.isReservedTag || no
  // first pass: mark all non-static nodes.
  markStatic(root)
  // second pass: mark static roots.
  markStaticRoots(root, false)
}

  首先我们可以看到这里他给了我们很长一段注释：
  告诉我们优化器的目标是遍历生成的AST树并检测纯静态的子树(即无需更改的DOM部分)。一旦检测到静态DOM树，我们就可以：
  1.将它们提升为常量，这样我们就不需要在每次重新渲染时为它们创建新节点；
  2.在patch过程中跳过它们。
  我们来看代码执行内容，首先接收了parse生成的AST，也就是root节点，如果root不存在的话就直接什么都不做。
  我们前面已经说了options.staticKeys就是'staticClass','staticStyle'这样的一个字符串，这里的isStaticKey是通过genStaticKeysCached方法传入这样一个字符串得到的。
  我们来看genStaticKeysCached是个什么

const genStaticKeysCached = cached(genStaticKeys)

  就是通过cached缓存了genStaticKeys函数

function genStaticKeys (keys: string): Function {
  return makeMap(
    'type,tag,attrsList,attrsMap,plain,parent,children,attrs,start,end,rawAttrsMap' +
    (keys ? ',' + keys : '')
  )
}

  可以看到这其实是通过makeMap方法又创建一个检查方法，并且将keys拼接到了后面，也就是后面加了',' + 'staticClass','staticStyle'这么一个字符串。所以这里的isStaticKey方法就是用来检查一个key是不是这里面的属性。
  接下来定义isPlatformReservedTag来判断是否是平台保留标签。
  然后接下来执行两个操作markStatic：标记非静态节点；markStaticRoots：标记静态根。
  我们先来看markStatic方法

function markStatic (node: ASTNode) {
  node.static = isStatic(node)
  if (node.type === 1) {
    // do not make component slot content static. this avoids
    // 1. components not able to mutate slot nodes
    // 2. static slot content fails for hot-reloading
    if (
      !isPlatformReservedTag(node.tag) &&
      node.tag !== 'slot' &&
      node.attrsMap['inline-template'] == null
    ) {
      return
    }
    for (let i = 0, l = node.children.length; i < l; i++) {
      const child = node.children[i]
      markStatic(child)
      if (!child.static) {
        node.static = false
      }
    }
    if (node.ifConditions) {
      for (let i = 1, l = node.ifConditions.length; i < l; i++) {
        const block = node.ifConditions[i].block
        markStatic(block)
        if (!block.static) {
          node.static = false
        }
      }
    }
  }
}

  首先会判断node是否是一个静态节点，我们来看一下isStatic方法

function isStatic (node: ASTNode): boolean {
  if (node.type === 2) { // expression
    return false
  }
  if (node.type === 3) { // text
    return true
  }
  return !!(node.pre || (
    !node.hasBindings && // no dynamic bindings
    !node.if && !node.for && // not v-if or v-for or v-else
    !isBuiltInTag(node.tag) && // not a built-in
    isPlatformReservedTag(node.tag) && // not a component
    !isDirectChildOfTemplateFor(node) &&
    Object.keys(node).every(isStaticKey)
  ))
}

  如果node.type为2其实就说明是有插值表达式，那显然不是静态的，所以直接返回false。如果是type为3就说明是文本节点或者注释节点，直接返回true。
  如果是其他情况，则如果使用v-pre指令则认为是一个静态节点。如果没有v-pre那就要满足!hasBindings，也就是不能有动态绑定的属性；还要!node.if和!node.for，也就是不能有v-if、v-for等指令；还要满足!isBuiltInTag等vue保留标签；然后是不能是平台保留标签；然后是!isDirectChildOfTemplateFor；最后是node的所有属性都要在isStaticKey列表中。满足以上所有条件才认为是一个静态节点。
我们来看isDirectChildOfTemplateFor方法

function isDirectChildOfTemplateFor (node: ASTElement): boolean {
  while (node.parent) {
    node = node.parent
    if (node.tag !== 'template') {
      return false
    }
    if (node.for) {
      return true
    }
  }
  return false
}

  如果node.parent存在，进入循环，首先将node设置为node.parent，如果其tag值不等于template就返回false，如果为template模板并且绑定了v-for才返回true。
  这是这个函数的理解，别忘了我们这里判断的是!isDirectChildOfTemplateFor，也就是说这里是排除绑定v-for的template标签。
  接下来再往下看markStatic，如果type为1的话，然后才会执行剩余逻辑。首先是判断不是一个平台保留标签，并且不是一个slot标签，并且不是内联模板就直接return。也就是说这些情况不能标记它们为静态子节点。如果说在组件标签内部写的插槽内容那是不能当成静态处理的。
  否则会进入下面for循环，然后其实是再遍历其子节点，并递归调用markStatic方法，并且如果子节点有非静态节点的话，也要把父节点设置为非静态。
  而对于使用v-if系列指令控制的节点，如果一组条件控制节点，如果有一组是非静态的，那么就把node.static赋值为false。
  其实markStatic整个过程就是一个深度遍历的过程，如果某个节点乃至它的子节点都是静态的，那么它就是个静态节点，否则为非静态节点。对于静态节点要打上static为true的标记。
  标记完静态节点后还要去标记静态根，也就是执行markStaticRoots方法

function markStaticRoots (node: ASTNode, isInFor: boolean) {
  if (node.type === 1) {
    if (node.static || node.once) {
      node.staticInFor = isInFor
    }
    // For a node to qualify as a static root, it should have children that are not just static text. 
    // Otherwise the cost of hoisting out will outweigh the benefits and it's better off to just always render it fresh.
    if (node.static && node.children.length && !(
      node.children.length === 1 &&
      node.children[0].type === 3
    )) {
      node.staticRoot = true
      return
    } else {
      node.staticRoot = false
    }
    if (node.children) {
      for (let i = 0, l = node.children.length; i < l; i++) {
        markStaticRoots(node.children[i], isInFor || !!node.for)
      }
    }
    if (node.ifConditions) {
      for (let i = 1, l = node.ifConditions.length; i < l; i++) {
        markStaticRoots(node.ifConditions[i].block, isInFor)
      }
    }
  }
}

  如果判断type为1的话，再判断node是一个静态节点或者含有v-once指令那就把node.staticInFor设置为isInFor，isInFor是传入的第二个参数，初始传入的是false。
  然后接下来会给含有子节点的静态节点给标记为静态根，但是有一种情况例外，那就是含有子节点，但是只有一个文本子节点。因为把只有单个文本节点的元素标记为静态根带来的成本将超过其收益，所以保持它总是更新的就可以了。 接下来则对其子节点递归调用markStaticRoots方法，这时候就要把node.for作为第二参数传入。也就是会判断节点在不在v-for循环中，只要在v-for中，那么后续的isInFor都会是true。
  最后同样对node.ifConditions中的元素处理。以上就是我们optimize的全部内容了，最后经过处理的AST就要进入接下来代码生成的环节了。

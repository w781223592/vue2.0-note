# 7.5 transition（下）

上一小节我们还遗留了一部分逻辑，接下来我们来看whenTransitionEnds函数

export function whenTransitionEnds (
  el: Element,
  expectedType: ?string,
  cb: Function
) {
  const { type, timeout, propCount } = getTransitionInfo(el, expectedType)
  if (!type) return cb()
}

  先会调用getTransitionInfo，然后返回对象中解构type、timeout、propCount，如果我们没有定义type也就是没有指定是过渡还是动画，那就直接调用cb并返回。
  接下来来看getTransitionInfo函数

export let transitionProp = 'transition'
export let animationProp = 'animation'

export function getTransitionInfo (el: Element, expectedType?: ?string): {
  type: ?string;
  propCount: number;
  timeout: number;
  hasTransform: boolean;
} {
  return {
    type,
    timeout,
    propCount,
    hasTransform
  }
}

这个函数也是比较长，我们拆开来分析，它最后会返回这样一个对象。

  const styles: any = window.getComputedStyle(el)
  // JSDOM may return undefined for transition properties
  const transitionDelays: Array<string> = (styles[transitionProp + 'Delay'] || '').split(', ')
  const transitionDurations: Array<string> = (styles[transitionProp + 'Duration'] || '').split(', ')
  const transitionTimeout: number = getTimeout(transitionDelays, transitionDurations)
  const animationDelays: Array<string> = (styles[animationProp + 'Delay'] || '').split(', ')
  const animationDurations: Array<string> = (styles[animationProp + 'Duration'] || '').split(', ')
  const animationTimeout: number = getTimeout(animationDelays, animationDurations)

  首先计算el的style属性对象，然后这一大堆逻辑就是分别获取transition-delay、transition-duation、animation-delay、animation-duation样式。并且要根据', '分割成数组。
然后通过getTimeout去计算时间

function getTimeout (delays: Array<string>, durations: Array<string>): number {
  /* istanbul ignore next */
  while (delays.length < durations.length) {
    delays = delays.concat(delays)
  }

  return Math.max.apply(null, durations.map((d, i) => {
    return toMs(d) + toMs(delays[i])
  }))
}

  如果delays长度小于durations，那么就让delays重复一遍自身元素。然后就会将delay + duration然后找到最长的那个过渡时间。

// Old versions of Chromium (below 61.0.3163.100) formats floating pointer numbers
// in a locale-dependent way, using a comma instead of a dot.
// If comma is not replaced with a dot, the input will be rounded down (i.e. acting
// as a floor function) causing unexpected behaviors
// Chromium的旧版本（低于61.0.3163.100）以依赖于区域设置的方式格式化浮动指针数字，
// 使用逗号而不是点。如果逗号没有被点取代，则输入将向下取整（即
// 作为floor function）导致意外行为

function toMs (s: string): number {
  return Number(s.slice(0, -1).replace(',', '.')) * 1000
}

toMs方法是去获取毫秒数，这里会做一个兼容。 这里就是去计算过渡或者动画时长，以便能做过渡和动画结束处理。再往下

  let type: ?string
  let timeout = 0
  let propCount = 0
  /* istanbul ignore if */
  if (expectedType === TRANSITION) {
    if (transitionTimeout > 0) {
      type = TRANSITION
      timeout = transitionTimeout
      propCount = transitionDurations.length
    }
  } else if (expectedType === ANIMATION) {
    if (animationTimeout > 0) {
      type = ANIMATION
      timeout = animationTimeout
      propCount = animationDurations.length
    }
  } else {
    timeout = Math.max(transitionTimeout, animationTimeout)
    type = timeout > 0
      ? transitionTimeout > animationTimeout
        ? TRANSITION
        : ANIMATION
      : null
    propCount = type
      ? type === TRANSITION
        ? transitionDurations.length
        : animationDurations.length
      : 0
  }
  const hasTransform: boolean =
    type === TRANSITION &&
    transformRE.test(styles[transitionProp + 'Property'])

  就是定义返回的四个属性。
  如果我们定义type为transition，那么进入if逻辑，当过渡时间大于0时为type赋值为transition、timeout赋值为transitionTimeout，propCount则是赋值为过渡效果的数量。
  如果定义type为animation，那么同样的处理办法。
  如果用户没有定义type，则去找过渡时间和动画时间长的那个做为timeout，并且根据时间长的那个为type赋值，最后根type值为propCount赋值为过渡数量或者动画数量。
  hasTransform则是去找有没有定义过渡效果的css名称，如果有的话才是true，如果没有实际上过渡也没有意义了。
  通过这个正则来判断

const transformRE = /\b(transform|all)(,|$)/

  这就是getTransitionInfo全部内容，我们回到whenTransitionEnds函数中，接下来会执行

  const event: string = type === TRANSITION ? transitionEndEvent : animationEndEvent

  let ended = 0
  const end = () => {
    el.removeEventListener(event, onEnd)
    cb()
  }
  const onEnd = e => {
    if (e.target === el) {
      if (++ended >= propCount) {
        end()
      }
    }
  }
  setTimeout(() => {
    if (ended < propCount) {
      end()
    }
  }, timeout + 1)
  el.addEventListener(event, onEnd)

  定义event，如果type === TRANSITION返回transitionEndEvent，否则返回animationEndEvent。

const TRANSITION = 'transition'
const ANIMATION = 'animation'
export let transitionEndEvent = 'transitionend'
export let animationEndEvent = 'animationend'

这里两个type常量定义，还有事件常量定义。

export const hasTransition = inBrowser && !isIE9
if (hasTransition) {

  if (window.ontransitionend === undefined &&

  if (window.onanimationend === undefined &&

  这里会对事件常量做兼容性修正。
  也就是接下来要监听这个事件，这是我们浏览器自带的一个能力，能够监听一个过渡或者动画的结束，除非它被移除。然后回调函数为onEnd，如果触发事件的是元素本身那么会++ended，并判断是否大于等于propCount，如果等于说明过渡或动画都执行完成，那就调用end函数，end函数则会移除事件监听并且调用cb。这里还有个定时器，定时为timeout + 1，如果我们不能正确监听到事件的话将会执行它，确保end被调用。
  接下来我们来看出场过渡动画的设计。

会执行一个leave函数

export function leave (vnode: VNodeWithData, rm: Function) {
  const el: any = vnode.elm

  // call enter callback now
  if (isDef(el._enterCb)) {
    el._enterCb.cancelled = true
    el._enterCb()
  }
//省略...
}

  和我们enter函数相差不大，略微有点改动，但是因为逻辑比较多我们分开来解析。
  这里发现el._enterCb函数还在，那说明入场的动画的回调还没有执行，但是已经执行了我们离场函数leave，那么首先是要先将入场动画给删掉，也就是调用入场动画回调函数_enterCb()，否则可能会影响我们离场动画。但是调用之前要先把它的cancelled 设置为true，那么在_enterCb中将会有不同的逻辑。这部分逻辑也是配合我们enter函数的，同样在enter函数开头也有这样类似的逻辑，我们当时没有去分析，但它们都差不多的。

  const data = resolveTransition(vnode.data.transition)
  if (isUndef(data) || el.nodeType !== 1) {
    return rm()
  }

  /* istanbul ignore if */
  if (isDef(el._leaveCb)) {
    return
  }
  const {

  const expectsCSS = css !== false && !isIE9

  const explicitLeaveDuration: any = toNumber(

      ? duration.leave
      : duration
  )
  if (process.env.NODE_ENV !== 'production' && isDef(explicitLeaveDuration)) {
    checkDuration(explicitLeaveDuration, 'leave', vnode)
  }
  const cb = el._leaveCb = once(() => {
    if (el.parentNode && el.parentNode._pending) {
      el.parentNode._pending[vnode.key] = null
    }
    if (expectsCSS) {
      removeTransitionClass(el, leaveToClass)
      removeTransitionClass(el, leaveActiveClass)
    }
    if (cb.cancelled) {
      if (expectsCSS) {
        removeTransitionClass(el, leaveClass)
      }
      leaveCancelled && leaveCancelled(el)
    } else {
      rm()
      afterLeave && afterLeave(el)
    }
    el._leaveCb = null
  })

这段逻辑和enter也是相差不大，我们没必要再过多分析。我们来看这三个变量

  afterLeave,
  leaveCancelled,
  delayLeave,

是不是很熟悉，这是我们在transition组件最后添加的几个钩子函数

  mergeVNodeHook(data, 'afterEnter', performLeave)
  mergeVNodeHook(data, 'enterCancelled', performLeave)
  mergeVNodeHook(oldData, 'delayLeave', leave => { delayedLeave = leave })

  再往下leave函数就会用到delayLeave，执行delayLeave后上面两个钩子函数才有意义，否则 delayedLeave为undefined

    delayLeave(performLeave)

  如果delayLeave存在那么说明是mode为'in-out'，那么执行delayLeave并且将performLeave传入，也就是会将delayedLeave赋值为performLeave，之后如果调用'afterEnter'和'enterCancelled'就会执行performLeave。
  如果delayLeave不存在就直接调用performLeave

  function performLeave () {
    // the delayed leave may have already been cancelled
    if (cb.cancelled) {
      return
    }
    // record leaving element
    if (!vnode.data.show && el.parentNode) {
      (el.parentNode._pending || (el.parentNode._pending = {}))[(vnode.key: any)] = vnode
    }
    beforeLeave && beforeLeave(el)
    if (expectsCSS) {
      addTransitionClass(el, leaveClass)
      addTransitionClass(el, leaveActiveClass)
      nextFrame(() => {
        removeTransitionClass(el, leaveClass)
        if (!cb.cancelled) {
          addTransitionClass(el, leaveToClass)
          if (!userWantsControl) {
            if (isValidDuration(explicitLeaveDuration)) {
              setTimeout(cb, explicitLeaveDuration)
            } else {
              whenTransitionEnds(el, type, cb)
            }
          }
        }
      })
    }
    leave && leave(el, cb)
    if (!expectsCSS && !userWantsControl) {
      cb()
    }
  }

  如果离场动画被取消就直接return，然后会调用beforLeave钩子。接下来的逻辑都和我们入场时动画相差不大。这里封装在一个函数里主要是为了做delayLeave的回调。这样才能保证入场动画执行完成后执行出场动。
这里有个逻辑

  离场动画执行前会将这个vnode通过key添加到父元素_pending对象中。然后在执行el._leaveCb时就会清空它。

  if (el.parentNode && el.parentNode._pending) {

在调用insert钩子时

  mergeVNodeHook(vnode, 'insert', () => {
    const parent = el.parentNode
    const pendingNode = parent && parent._pending && parent._pending[vnode.key]
    if (pendingNode &&
      pendingNode.tag === vnode.tag &&
      pendingNode.elm._leaveCb
    ) {

  如果发现parent._pending中有值，并且tag和新插入的元素tag相同，那么会先把旧的元素动画清除，这是防止快速切换插入和移除元素带来的bug。如果没有这个逻辑快速切换时会发生新元素插入但旧元素还没有消失，这样就会短时间同时出现两个元素影响体验。
  关于移除动画有个难点是要在动画结束后再去真正移除dom，这是怎么实现的呢？

  if (isDef(parentElm)) {
    removeVnodes([oldVnode], 0, 0)
  } else if (isDef(oldVnode.tag)) {
    invokeDestroyHook(oldVnode)
  }

  patch最后执行removeVnodes时会移除节点，但是里面有这样一段逻辑，我们一直不知道是什么意思，里面调用removeAndInvokeRemoveHook

  function removeAndInvokeRemoveHook(vnode, rm) {
    if (isDef(rm) || isDef(vnode.data)) {
      let i
      const listeners = cbs.remove.length + 1
      if (isDef(rm)) {
        rm.listeners += listeners
      } else {
        rm = createRmCb(vnode.elm, listeners)
      }
      // recursively invoke hooks on child component root node
      if (isDef(i = vnode.componentInstance) && isDef(i = i._vnode) && isDef(i.data)) {
        removeAndInvokeRemoveHook(i, rm)
      }
      for (i = 0; i < cbs.remove.length; ++i) {
        cbs.remove[i](vnode, rm)
      }
      if (isDef(i = vnode.data.hook) && isDef(i = i.remove)) {
        i(vnode, rm)
      } else {
        rm()
      }
    } else {
      removeNode(vnode.elm)
    }
  }

  移除节点时因我们节点会添加data.transition，所以这里会执行if逻辑。else逻辑是会直接删除dom，这里是不会执行else逻辑的。
  if逻辑中获取cbs.remove.length，其实就transition中的remove钩子，这里rm不存在的话会执行else逻辑，也就是createRmCb，if逻辑是有嵌套处理的复杂情况，我们就没必要去分析了。createRmCb函数会传入listeners，也就是remove函数数量加1，这里为什么加1我们先留个悬念。

  function createRmCb(childElm, listeners) {
    function remove() {
      if (--remove.listeners === 0) {
        removeNode(childElm)
      }
    }
    remove.listeners = listeners
    return remove
  }

  这里我们remove是只有一个的，所以这里listeners为2，然后createRmCb定义了remove函数，然后将listeners赋值给remove.listeners，最后返回remove。我们先分析一下remov做了什么事，之后我们再分析它什么时候会调用。
  它会先将listeners减1，然后判断是否为0，如果为0就调用removeNode真正去移除dom。
  回到removeAndInvokeRemoveHook，接下来会对子组件递归调用此方法，这时会传入rm，也就可能会发生我们上述初始rm不为空的复杂情况，这种情况我们不去讨论。接下来会调用cbs.remove中的函数，也就是执行我们leave逻辑并且会传入rm。rm是在el._leaveCb执行的，也就是离场后回调中。但是el._leaveCb是异步执行的，再此之前会先执行removeAndInvokeRemoveHook函数。
  然后会执行else逻辑去调用一次rm，我们知道rm调用时会使rm.listeners减1，所以这里它就从2变为1。也就是下一次调用时它就会变为0，从而移除我们dom。那么下一次调用是在多会呢？就是在我们el._leaveCb回调中，这时候离场动画执行完毕，那我们也就能将它删除了。
  以上就是我们transition组件的全部内容了。实际上我们可以看到transition没有任何过渡或动画逻辑，而是去给我们过渡或动画的class或者js做一个管理，从而让我们更高效的开发动效。

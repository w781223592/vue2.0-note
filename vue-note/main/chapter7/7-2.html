# 7.2 keep-alive（下）

  上个小节我们分析了keep-alive组件的实现和更新过程。那么接下来我们分析由keep-alive组件引入的两个新的生命周期activated和deactivated。
  在patch方法最后会执行invokeInsertHook方法

  function invokeInsertHook(vnode, queue, initial) {
    // delay insert hooks for component root nodes, invoke them after the
    // element is really inserted 延迟为组件根节点插入钩子，在真正插入元素后调用它们
    if (isTrue(initial) && isDef(vnode.parent)) {
      vnode.parent.data.pendingInsert = queue
    } else {
      for (let i = 0; i < queue.length; ++i) {
        queue[i].data.hook.insert(queue[i])
      }
    }
  }

  我们之前已经分析过，这里其实会执行insert钩子，来确保调用mounted钩子在dom真正插入时调用。我们看insert还执行了哪些逻辑

  insert (vnode: MountedComponentVNode) {
    const { context, componentInstance } = vnode
    if (!componentInstance._isMounted) {
      componentInstance._isMounted = true
      callHook(componentInstance, 'mounted')
    }
    if (vnode.data.keepAlive) {
      if (context._isMounted) {
        // vue-router#1212
        // During updates, a kept-alive component's child components may
        // change, so directly walking the tree here may call activated hooks on incorrect children. Instead we push them into a queue which willbe processed after the whole patch process ended.
        queueActivatedComponent(componentInstance)
      } else {
        activateChildComponent(componentInstance, true /* direct */)
      }
    }
  },

  这里判断如果vnode.data.keepAlive存在的话，并且context已经mounted的话，context其实就是它父组件实例。会执行queueActivatedComponent。这个是解决一个路由的bug，在更新期间如果缓存的子组件可能发生变化。所以这里直接遍历去调用的话可能会激活错误的组件钩子，所以把它推到一个队列，在整个patch结束后再去处理。

export function queueActivatedComponent(vm: Component) {
  // setting _inactive to false here so that a render function can
  // rely on checking whether it's in an inactive tree (e.g. router-view)
  vm._inactive = false
  activatedChildren.push(vm)
}

  这里将vm._inactive赋值为false，然后将它添加到activatedChildren数组，在flushSchedulerQueue调度函数执行完所有异步队列后再去执行这两个逻辑

  const activatedQueue = activatedChildren.slice()
  const updatedQueue = queue.slice()

  resetSchedulerState()

  // call component updated and activated hooks
  callActivatedHooks(activatedQueue)
  callUpdatedHooks(updatedQueue)

  这个执行我们应该是没有去分析的，这里获取到activatedChildren的副本然后就会调用resetSchedulerState将其重置，然后会执行callActivatedHooks

function callActivatedHooks(queue) {
  for (let i = 0; i < queue.length; i++) {
    queue[i]._inactive = true
    activateChildComponent(queue[i], true /* true */)
  }
}

  遍历并将其_inactive设置为true，然后调用activateChildComponent

export function activateChildComponent(vm: Component, direct?: boolean) {
  if (direct) {
    vm._directInactive = false
    if (isInInactiveTree(vm)) {
      return
    }
  } else if (vm._directInactive) {
    return
  }
  if (vm._inactive || vm._inactive === null) {
    vm._inactive = false
    for (let i = 0; i < vm.$children.length; i++) {
      activateChildComponent(vm.$children[i])
    }
    callHook(vm, 'activated')
  }
}

  这里传入第二个参数direct为true，这里如果direct为true那么就把vm._directInactive置为false，如果判断这里激活的组件在一不活跃的树上时就不做任何操作。

function isInInactiveTree(vm) {
  while (vm && (vm = vm.$parent)) {
    if (vm._inactive) return true
  }
  return false
}

  isInInactiveTree其实就向上寻找组父组件实例，如果找到vm._inactive为true的话就返回true。这时候会选择不去调用激活钩子，因为父组件未激活，那么子组件也不应该被激活，因为这里调用激活钩子是异步的，中间可能有地方再次改变了一些状态，所以要设置一些标识去处理。
  接下来我们会看到如果vm._inactive为true，那么会将其置为false并且递归去调用activateChildComponent处理子组件，也就是说父组件变为活跃前先会递归处理子组件，如果子组件_inactive为true的话，就会先调用子组件的activated钩子。
  所以之前会检查父组件是否有将要被激活的，因为父组件会处理子组件，如果发现父组件未激活，那就让父组件去处理就好了。这样是有利于处理activated调用顺序的。
  至于else if中_directInactive是组件失活时的处理标识，所以这里发现它为true也不会再往下执行激活逻辑。
  我们再看销毁时逻辑，在patch方法最后会移除旧节点并销毁旧组件。

  // destroy old node
  if (isDef(parentElm)) {
    removeVnodes([oldVnode], 0, 0)
  } else if (isDef(oldVnode.tag)) {
    invokeDestroyHook(oldVnode)
  }

  执行removeVnodes方法

  function removeVnodes(vnodes, startIdx, endIdx) {
    for (; startIdx <= endIdx; ++startIdx) {
      const ch = vnodes[startIdx]
      if (isDef(ch)) {
        if (isDef(ch.tag)) {
          removeAndInvokeRemoveHook(ch)
          invokeDestroyHook(ch)
        } else { // Text node
          removeNode(ch.elm)
        }
      }
    }
  }

  这个方法又会执行invokeDestroyHook

  function invokeDestroyHook(vnode) {
    let i, j
    const data = vnode.data
    if (isDef(data)) {
      if (isDef(i = data.hook) && isDef(i = i.destroy)) i(vnode)
      for (i = 0; i < cbs.destroy.length; ++i) cbs.destroy[i](vnode)
    }
    if (isDef(i = vnode.children)) {
      for (j = 0; j < vnode.children.length; ++j) {
        invokeDestroyHook(vnode.children[j])
      }
    }
  }

  如果vnode.data存在然后会判断data.hook.destroy是否存在，如果存在就会执行destroy钩子

  destroy (vnode: MountedComponentVNode) {
    const { componentInstance } = vnode
    if (!componentInstance._isDestroyed) {
      if (!vnode.data.keepAlive) {
        componentInstance.$destroy()
      } else {
        deactivateChildComponent(componentInstance, true /* direct */)
      }
    }
  }

  如果组件不是被销毁状态，然后就会执行内部逻辑，如果这个时候keepAlive为true的话就会执行deactivateChildComponent

export function deactivateChildComponent(vm: Component, direct?: boolean) {
  if (direct) {
    vm._directInactive = true
    if (isInInactiveTree(vm)) {
      return
    }
  }
  if (!vm._inactive) {
    vm._inactive = true
    for (let i = 0; i < vm.$children.length; i++) {
      deactivateChildComponent(vm.$children[i])
    }
    callHook(vm, 'deactivated')
  }
}

  这里先把directInactive 置为true，然后再去找组件是否在未激活的树上，如果在的话也不需要去调用失活钩子，接下的逻辑就激活差不多，就是判断!vm._inactive时去调用deactivated钩子，同样的如果子组件也要递归去调用失活钩子。

# 4.10 diff算法

  diff算法相信我们都不会陌生，只要使用过Vue都或多或少听过它，它是Vue一个比较核心的概念。diff即difference（不同）的缩写，它所做的工作就是去找不同。我们知道dom对象是非常庞大的，学完上一小节后我们知道Vue更新实际上是以组件为单位的，一个组件会包含许多子节点，如果每次更新都重新创建新的元素那无疑开销是十分大的。为了解决这个问题就有了diff算法。   那么diff算法做了哪些事呢？在更新过程中它会重新执行render函数生成新的vnode，通过对比新旧节点不同来确定那些节点需要重新创建，哪些节点需要更新，哪些节点需要移动、哪些节点需要移除。也就是说把重复创建dom节点的过程转化为重复创建虚拟dom，这个过程开销就小的多，通过对比新旧vnode，就可以知道可以复用哪些dom，不能复用的就销毁，新的节点就重新创建，复用的节点可以做内容更新，如果定义了key值，一般我们在v-for循环中会定义唯一key值，这样通过相同的key就可以确定可复用节点，这样如果节点位置发生变化就可以找到新节点的位置去移动dom。
  接下来我们分析Vue是怎么实现这个过程的，Vue的diff算法我之前已经接触过了，就是我们在patch过程中执行的patchVnode方法，对于patchVnode方法我们上个小节已经分析过了，在patchVnode中有一个updateChildren方法，这个方法就是我们接下来分析的重点内容，因为它是diff算法的核心。

function updateChildren(parentElm, oldCh, newCh, insertedVnodeQueue, removeOnly) {
  let oldStartIdx = 0
  let newStartIdx = 0
  let oldEndIdx = oldCh.length - 1
  let oldStartVnode = oldCh[0]
  let oldEndVnode = oldCh[oldEndIdx]
  let newEndIdx = newCh.length - 1
  let newStartVnode = newCh[0]
  let newEndVnode = newCh[newEndIdx]
  let oldKeyToIdx, idxInOld, vnodeToMove, refElm

  // removeOnly is a special flag used only by <transition-group>
  // to ensure removed elements stay in correct relative positions
  // during leaving transitions
  // removeOnly是一个特殊标志，仅由＜transition group＞使用，
  // 以确保在离开转换期间被移除的元素保持在正确的相对位置
  const canMove = !removeOnly

  if (process.env.NODE_ENV !== 'production') {
    checkDuplicateKeys(newCh)
  }

  while (oldStartIdx <= oldEndIdx && newStartIdx <= newEndIdx) {
    if (isUndef(oldStartVnode)) {
      oldStartVnode = oldCh[++oldStartIdx] // Vnode has been moved left
    } else if (isUndef(oldEndVnode)) {
      oldEndVnode = oldCh[--oldEndIdx]
    } else if (sameVnode(oldStartVnode, newStartVnode)) {
      patchVnode(oldStartVnode, newStartVnode, insertedVnodeQueue, newCh, newStartIdx)
      oldStartVnode = oldCh[++oldStartIdx]
      newStartVnode = newCh[++newStartIdx]
    } else if (sameVnode(oldEndVnode, newEndVnode)) {
      patchVnode(oldEndVnode, newEndVnode, insertedVnodeQueue, newCh, newEndIdx)
      oldEndVnode = oldCh[--oldEndIdx]
      newEndVnode = newCh[--newEndIdx]
    } else if (sameVnode(oldStartVnode, newEndVnode)) { // Vnode moved right
      patchVnode(oldStartVnode, newEndVnode, insertedVnodeQueue, newCh, newEndIdx)
      canMove && nodeOps.insertBefore(parentElm, oldStartVnode.elm, nodeOps.nextSibling(oldEndVnode.elm))
      oldStartVnode = oldCh[++oldStartIdx]
      newEndVnode = newCh[--newEndIdx]
    } else if (sameVnode(oldEndVnode, newStartVnode)) { // Vnode moved left
      patchVnode(oldEndVnode, newStartVnode, insertedVnodeQueue, newCh, newStartIdx)
      canMove && nodeOps.insertBefore(parentElm, oldEndVnode.elm, oldStartVnode.elm)
      oldEndVnode = oldCh[--oldEndIdx]
      newStartVnode = newCh[++newStartIdx]
    } else {
      if (isUndef(oldKeyToIdx)) oldKeyToIdx = createKeyToOldIdx(oldCh, oldStartIdx, oldEndIdx)
      idxInOld = isDef(newStartVnode.key)
        ? oldKeyToIdx[newStartVnode.key]
        : findIdxInOld(newStartVnode, oldCh, oldStartIdx, oldEndIdx)
      if (isUndef(idxInOld)) { // New element
        createElm(newStartVnode, insertedVnodeQueue, parentElm, oldStartVnode.elm, false, newCh, newStartIdx)
      } else {
        vnodeToMove = oldCh[idxInOld]
        if (sameVnode(vnodeToMove, newStartVnode)) {
          patchVnode(vnodeToMove, newStartVnode, insertedVnodeQueue, newCh, newStartIdx)
          oldCh[idxInOld] = undefined
          canMove && nodeOps.insertBefore(parentElm, vnodeToMove.elm, oldStartVnode.elm)
        } else {
          // same key but different element. treat as new element
          createElm(newStartVnode, insertedVnodeQueue, parentElm, oldStartVnode.elm, false, newCh, newStartIdx)
        }
      }
      newStartVnode = newCh[++newStartIdx]
    }
  }
  if (oldStartIdx > oldEndIdx) {
    refElm = isUndef(newCh[newEndIdx + 1]) ? null : newCh[newEndIdx + 1].elm
    addVnodes(parentElm, refElm, newCh, newStartIdx, newEndIdx, insertedVnodeQueue)
  } else if (newStartIdx > newEndIdx) {
    removeVnodes(oldCh, oldStartIdx, oldEndIdx)
  }
}

  可以看到这部分代码是非常庞大的，我们先从参数来分析它。第一个参数parentElm：即父节点，是一个真实dom；接下来两个参数分别是新旧vnode子节点队列，后两个参数我们可以暂时不用关注，removeOnly可以认为它是false。   vue2中的diff算法是双端比较的，它通过四个指针来比较，四个指针分别是新旧vnode数组的第一位和最后一位的索引。通过四个索引找到四个对应vnode，也就是我们如下逻辑：

  let oldStartIdx = 0
  let newStartIdx = 0
  let oldEndIdx = oldCh.length - 1
  let oldStartVnode = oldCh[0]
  let oldEndVnode = oldCh[oldEndIdx]
  let newEndIdx = newCh.length - 1
  let newStartVnode = newCh[0]
  let newEndVnode = newCh[newEndIdx]
  let oldKeyToIdx, idxInOld, vnodeToMove, refElm

  最后一句定义四个空变量，后面使用时再来看它们做了什么事。
  接下来

  // removeOnly is a special flag used only by <transition-group>
  // to ensure removed elements stay in correct relative positions
  // during leaving transitions
  // removeOnly是一个特殊标志，仅由＜transition group＞使用，
  // 以确保在离开转换期间被移除的元素保持在正确的相对位置
  const canMove = !removeOnly

  if (process.env.NODE_ENV !== 'production') {
    checkDuplicateKeys(newCh)
  }

  定义canMove为!removeOnly，注释说的比较清楚，removeOnly是专门为动画组件定义的标识。
  在开发环境执行checkDuplicateKeys(newCh)，这方法是用来检查是否有重复key值的，我们之前已经分析过了。

  再往下是一个大大的while循环，循环条件是oldStartIdx <= oldEndIdx && newStartIdx <= newEndIdx，我们以上图做参考，假设现在有list=[A,B,C,D]，通过v-for渲染到页面，我们以各自内容作为key，Vue会生成一个vnode队列oldCh。然后我们修改list为[C,A,D,B]，然后又生成新的vnode的队列newCh，接下来patch过程就通过上面逻辑去对比新旧节点来更新dom，那接下来我们就看看执行过程中会做哪些事。
  我们可以看到while循环中有7个分支，我们先来看3、4、5、6这四个分支，因为它们原理比较相近，也就是这四个指针两两相互比较，分别是：oldStartVnode — newStratVnode、oldEndVnode — newEndVnode、oldStartVnode — newEndVnode、oldEndVnode — newStartVnode。（只在新旧vnode两组间比较）
  下面我们就按这个顺序来分析，经过上述对比，我们发现没有一对字母是相同的，所以不存在相同节点，那这样我们这四个分支就都不会执行，但是我们可以清楚的看到，新旧vnode只是数据顺序变化了，其他部分都没有改变。最理想的情况下就是我们这四个分支判断成立，但是这里它并没有命中任何一个分支判断，说明我们还应该增加一个else分支，也就是我们最后一个分支。
  现在我们就来分析else的内容：

  else {
    if (isUndef(oldKeyToIdx)) oldKeyToIdx = createKeyToOldIdx(oldCh, oldStartIdx, oldEndIdx)
    idxInOld = isDef(newStartVnode.key)
      ? oldKeyToIdx[newStartVnode.key]
      : findIdxInOld(newStartVnode, oldCh, oldStartIdx, oldEndIdx)
    if (isUndef(idxInOld)) { // New element
      createElm(newStartVnode, insertedVnodeQueue, parentElm, oldStartVnode.elm, false, newCh, newStartIdx)
    } else {
      vnodeToMove = oldCh[idxInOld]
      if (sameVnode(vnodeToMove, newStartVnode)) {
        patchVnode(vnodeToMove, newStartVnode, insertedVnodeQueue, newCh, newStartIdx)
        oldCh[idxInOld] = undefined
        canMove && nodeOps.insertBefore(parentElm, vnodeToMove.elm, oldStartVnode.elm)
      } else {
        // same key but different element. treat as new element
        createElm(newStartVnode, insertedVnodeQueue, parentElm, oldStartVnode.elm, false, newCh, newStartIdx)
      }
    }
    newStartVnode = newCh[++newStartIdx]
  }

  首先判断oldKeyToIdx是undefined的话，就执行createKeyToOldIdx函数，并将返回结果赋值给oldKeyToIdx。

function createKeyToOldIdx(children, beginIdx, endIdx) {
  let i, key
  const map = {}
  for (i = beginIdx; i <= endIdx; ++i) {
    key = children[i].key
    if (isDef(key)) map[key] = i
  }
  return map
}

  这个方法主要是创建旧vnode的key与索引idx的映射表并返回，数组遍历的范围就根据传入的oldStartIdx和oldEndIdx来取的。
  然后给idxInOld赋值，赋值表达式是一个三目运算，当newStartVnode.key存在时，则取oldKeyToIdx[newStartVnode.key]，否则执行findIdxOld方法：

  function findIdxInOld(node, oldCh, start, end) {
    for (let i = start; i < end; i++) {
      const c = oldCh[i]
      if (isDef(c) && sameVnode(node, c)) return i
    }
  }

  findIdxInOld方法就是从oldStartIdx到oldEndIndx位置遍历oldCh，找到与newStartVnode相同的节点就返回它的索引。由此可见如果我们不传入key值时就会执行遍历逻辑，造成性能的浪费。这种嵌套的循环会造成时间复杂度的提升，而定义key之后可以通过映射表oldKeyToIdx去找相同节点，这是一种时间转空间来降低时间复杂度的思想。
  接下来如果发现idxInOld是undefined，说明没有可共用的节点，那么就调用createElm方法去创建新节点。idxInOld有值就使vnodeToMove=oldCh[idxInOld]，然后通过sameVnode检查是否是同一节点，因为key相同并不代表一定是相同节点，这里所有的相同节点都要警告sameVnode的检验。
  如果sameVnode检验通过则通过patchVnode方法来更新节点就好了，这是个递归调用的过程，如果vnode含有children还会重复此过程。当然这还没有完，我们只是更新了子节点内容，像我们图示那样我们newStartVnode是C然后匹配到oldCh中相同节点的索引是2，然后我们执行patchVnode后会更新旧节点，但是我们发现新旧节点的内容都是C，所以这里执行patchVnode后实际还保持原样。这样我们发现目前C的索引位置还是在2号，但是在新的数组中它应该是0号位置，说明我们光进行更新内容还不够，还需要调整dom的位置。
  所以接下来就是调整dom位置的逻辑了，首先我们把oldCh[idxInOld]置为undefined，因为它已经被vnodeToMove保存了。然后canMove为true的话执行插入方法insertBefore，插入位置为父节点parentElm下，参考节点oldStartVnode.elm前，也就是把C插入A位置前，这样C就和更新后的结果表现一致了。
  如果sameVnode检验不通过那就还是去创建新节点处理，这种情况下创建节点后就直接插入到父节点下的正确位置了。这两个分支结束最后要重新赋值newStartVnode，因为newCh中第一个元素已经处理完了，我们就需要处理第二个元素的更新了，同时指针newStartIdx也需要加1。

  直接看我们效果图，第一次循环后我们变成了上图，左边第一个C用方框'框'起来是因为它不在oldCh中，这里表示的是一个目前真实dom的位置。这时候oldCh第三个vnode已经变成了undefined。
  然后继续执行while循环，这时就执行到我们之前说的四个分支代码了，显然这时候oldStartVnode和newStratVnode是相同节点，因为都指向A，也就是我们第3分支的代码：

      else if (sameVnode(oldStartVnode, newStartVnode)) {
        patchVnode(oldStartVnode, newStartVnode, insertedVnodeQueue, newCh, newStartIdx)
        oldStartVnode = oldCh[++oldStartIdx]
        newStartVnode = newCh[++newStartIdx]
      }

  这个时候位置是相同的，所以我们不需要移动节点，只需要patchVnode就可以了，之后再把oldStartIdx和newStartIdx指针后移并更新对应的oldStartVnode和newStratVnode。关于移动指针，我们这里是处理了哪个节点就移动哪个指针，至于前移还是后移就看是起始指针还是末尾指针了。

  这时再次执行while循环，oldStartVnode和newEndVnode是相同节点，这就执行到了第5分支：

  else if (sameVnode(oldStartVnode, newEndVnode)) { // Vnode moved right
    patchVnode(oldStartVnode, newEndVnode, insertedVnodeQueue, newCh, newEndIdx)
    canMove && nodeOps.insertBefore(parentElm, oldStartVnode.elm, nodeOps.nextSibling(oldEndVnode.elm))
    oldStartVnode = oldCh[++oldStartIdx]
    newEndVnode = newCh[--newEndIdx]
  }

  这里除了要做patchVnode外还需要把旧的B节点后移，这里移动节点仍然是使用node.insertBefore，这里是要把B节点插到D节点后面，所以这里是去找D节点的后一个兄弟节点作为参考节点，然后将B插入D的nextSiling节点前面。这里D明显没有后一位兄弟节点了，所以node.nextSiling为null，正好insertBefore的特性就是如果参考节点为null则插入到父元素最后一位。所以是完全符合要求的。
  最后oldStartIdx后移，newEndIdx前移，然后进入下一个循环。

  这时候newStartVnode和newEndVnode重合，但是还满足循环条件，这时候我们发现oldStartVnode为undefined，这时候如果sameVnode对比的话就会出错了，因为undefined不能去读取属性。处理方式也非常简单，在循环开头加上两个判断逻辑就行了，如果遇到undefined跳过就行了，也就是我们第1、2分支逻辑。这里走的是第1分支逻辑：

  if (isUndef(oldStartVnode)) {
    oldStartVnode = oldCh[++oldStartIdx] // Vnode has been moved left
  }

  oldStartIdx后移，这里我们就不专门去列出它的图了，这时oldStartVnode和oldEndVnode也相同了，这时候已经到了循环的边界条件了。正好两个D节点是相同的，然后再次执行第三分支就可以了，到这里我们这个大while循环就分析完毕了。等等，我们好像没有分析第2、4、6分支的内容。其实它和1、3、5分支逻辑类似，这里就不去赘述了。
  但是我们上面的例子只分析了顺序不同的情况，很多时候新旧vnode长度也是不一样的。假设我们oldCh的C元素和D元素中间多一个C+元素，那么跳出循环时C+元素并没有被处理，因为这时newEndIdx < newStartIdx了。那么我们需要额外的逻辑来处理这种情况。

  如图所示，稍稍修改上面例子，下次循环时候会因为不满足循环条件而退出循环这时显然C+我们还没去处理。所以循环之后还有一段逻辑：

  if (oldStartIdx > oldEndIdx) {
    refElm = isUndef(newCh[newEndIdx + 1]) ? null : newCh[newEndIdx + 1].elm
    addVnodes(parentElm, refElm, newCh, newStartIdx, newEndIdx, insertedVnodeQueue)
  } else if (newStartIdx > newEndIdx) {
    removeVnodes(oldCh, oldStartIdx, oldEndIdx)
  }

  这种情况属于newCh已经添加完了，但是旧节点还没有遍历完全，说明旧节点还有多余部分，也就是走else if分支逻辑，调用removeVnodes将多余节点移除就好了。移除方法我们在前面已经分析过了，这里也没必要再去赘述了。再看if逻辑，也就是newCh还未处理完，处理方式也十分简单，将newStartIdx和newEndIdx之间的vnode节点全部插入就好了，但是插入前需要找到准确的位置，也就是找一个参考节点，于是就取newEndIdx的dom节点做为参考节点，然后调用addVnodes方法去插入。addVnodes方法就是遍历数组依次调用createElm方法。 到这里我们的diff算法就分析完了，最后思考一个问题，为什么最好不要用index作为key值？
